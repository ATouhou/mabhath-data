<درست>: ( دُرُسْتُ ، بضمَّتيْنِ ) وسكون : أَهمله الجماعةُ .
ودُرُسْت ( بنُ رِباطٍ ) ، ككِتاب .
( الفُقَيمِيُّ : شاعِرٌ ، وابْنُه زِيادٌ ) ، هاكذا في النُّسخ ، والصوابُ : وابنُ زِيادٍ ، كُنيتُه أَبو الحسنِ ويُقال : أَبو يحيى نفاض الخَزْرَوِيّ عن جعفر بن الزُّبَيْر ، وعليّ بن زيدِ بن جُدْعَانَ ، وعنه أَبو كاملٍ الجَحْدَرِيُّ ، وغيرُه ، كذا في حاشية الإِكمال ، يقال هو ضعِيفٌ .
وقال أَبو زُرْعَة : واهٍ .
( وابْنُهُ يَحْيَى ) بْنُ دُرُسْت بنِ زِيادِ شيخُ التِّرْمِذِيّ والنَّسائيّ .
( وابْنُ ابْنِهِ زَكرِيَّا ) بنُ يَحْيَى بنِ دُرُسْت بنِ زيادٍ ، عن هِشامِ بن عَمَّار ، وغيرِه .
( و ) دُرُسْتُ بن حَمزة ، و ( ابْنُ حَكِيمٍ ) ، مُكَبَّراً ، يَرْوِي عن التابِعين .
( و ) دُرُسْتُ ( بنُ سَهْلٍ ) ، عن سَهْلِ بنِ عُثْمانَ العَسكريّ .
( و ) دُرُسْتُ ( بْنُ نَصْرٍ الزاهِدُ ) مات سنة 241 ، وهو شيخٌ لاِبْنِ مخلد .
( وإِبراهيمُ بنُ جَعْفرِ بنِ دُرُسْت ) التُّسْتَرِيّ .
شيخٌ لاِبن المقري .
[ ] وفاته :
دُرُسْتُ بنُ حَمْزة عن مَطرٍ الوَرّاق ، قال الدّارَقُطْنِيُّ : ضعيفٌ .
ودُرُسْتُ عن أَبي أَيُّوبَ ثِقةٌ .
ودُرُسْتُ بنُ اللَّجْلاج العَبْدِيّ ، عن رَوْح بن عبد المُؤْمن .
( وجَعْفرُ بنُ دُرُسْتَوَيْهِ ) عن ابن المَدِينيّ ، وابنه أَبو محمّد عبد الله بن جعفر ، رَوى عن يعقوبَ بنِ سُفيانَ الفَسَويّ ( مُحدِّثُون ) .
وأَبو أَحمد عبدُ الحميد بن محمّد بن الحسين بن عبد الله السّمْسار الدُّرُسْتويّ ، لأَنّ جدّه عُرِف بابن غُلامِ دُرُسْتَوَيْه ، بَلْخِيُّ الأَصلِ ، سكنَ بَغدادَ ، وروى عن لُوَيْنٍ وغيرِه .
( تُوُفِّيَ سنة 318 ) .
