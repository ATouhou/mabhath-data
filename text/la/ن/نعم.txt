صَبَّحكَ اللهُ بخَيْرٍ باكرِ،<br>\

أَشمُّ من هَيْقٍ وأَهْدَى من جَمَلْ<br>\

نَعامةٍ. ويقال: ركب فلانٌ جَناحَيْ نَعامةٍ إذا جدَّ في أَمره. ويقال<br>\

فأَما بنو عامرٍ بالنِّسار<br>\

نَعامَتُهم، وخَفَّتْ نَعامَتُهم أَي استَمر بهم السيرُ. ويقال للعَذارَى:<br>\
كأنهن بَيْضُ نَعامٍ. ويقال للفَرَس: له ساقا نَعامةٍ لِقِصَرِ ساقَيْه،<br>\
وله جُؤجُؤُ نَعامةٍ لارتفاع جُؤْجُؤها. ومن أَمثالهم: مَن يَجْمع بين<br>\

ما أَنت إلا نَعامةٌ؛ يَعْنون قوله:<br>\
ومِثْلُ نَعامةٍ تُدْعَى بعيراً،<br>\

النعامة ذهَبَتْ تَطْلُبُ قَرْنَينِ فقطعوا أُذُنيها فجاءت بلا أُذُنين؛<br>\

أو كالنَّعامةِ، إذ غَدَتْ من بَيْتِها<br>\

غُصَّتَها وأَفْلَتَتْ، وبَقِيَت المرأَةُ لا صَيْدَها أَحْرَزَتْ ولا<br>\

الثِّقةِ. والنَّعامة: الخشبة المعترضة على الزُّرنُوقَيْنِ تُعَلَّق منهما<br>\

النَّعامتانِ خَشَبتين يُضَمُّ طرَفاهما الأَعْليان ويُرْكَز طرفاهما الأَسفلان في<br>\

اللتان عليهما الخشبة المعترِضة؛ وقال اللحياني: النَّعامتان الخشبتان اللتان<br>\

البئر تَقوم عليها السَّواقي. والنَّعامة: صخرة ناشزة في البئر.<br>\

بِهنَّ نَعامٌ بَناها الرجا<br>\

لا ظِلَّ في رَيْدِها<br>\

اشْرَبْ هنِيئاً فقد شالَتْ نَعامتُهم،<br>\

إني قَضَيْتُ قضاءً غيرَ ذي جَنَفٍ،<br>\
لَمَّا سَمِعْتُ ولمّا جاءَني الخَبَرُ<br>\

ولو أَنيّ حَدَوْتُ به ارْفَأَنَّتْ<br>\
نَعامتُه، وأَبْغَضَ ما أَقولُ<br>\

الأزهري: زعموا أَن ابن النعامة من الطرق كأَنه مركب النَّعامة من قوله:<br>\
وابن النعامة، يوم ذلك، مَرْكَبي<br>\
وابن النَعامة: الساقي الذي يكون على البئر. والنعامة: الرجْل.<br>\
والنعامة: الساق. والنَّعامة: الفَيْجُ المستعجِل. والنَّعامة: الفَرَح.<br>\

الظَّبْي، وجاؤوا على بَكْرة أَبيهم، وليس ثم داء ولا بَكرة. قال ابن<br>\
بري: وهذا البيت، أَعني فيكون مركبكِ، لِخُزَزَ بن لَوْذان السَّدوسيّ؛<br>\

كذَبَ العَتيقُ وماءُ شَنٍّ بارِدٍ،<br>\

وحُمْلانٍ، والعرب إذا أَفردت النَّعَم لم يريدوا بها إلا الإبل، فإذا قالوا<br>\
الأنعام أَرادوا بها الإبل والبقر والغنم، قال الله عز وجل: ومن<br>\
الأَنْعام حَمولةً وفَرْشاً كلوا مما رزقكم الله (الآية) ثم قال: ثمانية أَزواج؛<br>\
أي خلق منها ثمانية أَزواج، وكان الكسائي يقول في قوله تعالى: نسقيكم<br>\
مما في بطونه؛ قال: أَراد في بطون ما ذكرنا؛ ومثله قوله:<br>\
مِثْل الفراخ نُتِفَتْ حَواصِلُهْ<br>\
أي حواصل ما ذكرنا؛ وقال آخر في تذكير النَّعَم:<br>\
في كلّ عامٍ نَعَمٌ يَحْوونَهُ،<br>\
يُلْقِحُه قَوْمٌ ويَنْتِجونَهُ<br>\
ومن العرب من يقول للإبل إذا ذُكِرت <br>\
(* قوله «إذا ذكرت» الذي في<br>\
التهذيب: كثرت) الأَنعام والأَناعيم.<br>\
والنُّعامى، بالضم على فُعالى: من أَسماء ريح الجنوب لأَنها أَبلُّ<br>\
الرياح وأَرْطَبُها؛ قال أَبو ذؤيب:<br>\
مَرَتْه النُّعامى فلم يَعْتَرِفْ،<br>\
خِلافَ النُّعامى من الشَّأْمِ، ريحا<br>\
وروى اللحياني عن أَبي صَفْوان قال: هي ريح تجيء بين الجنوب والصَّبا.<br>\
والنَّعامُ والنَّعائمُ: من منازل القمر ثمانيةُ كواكبَ: أَربعة صادرٌ،<br>\
وأَربعة واردٌ؛ قال الجوهري: كأنها سرير مُعْوجّ؛ قال ابن سيده: أَربعةٌ<br>\
في المجرّة وتسمى الواردةَ وأَربعة خارجة تسمَّى الصادرةَ. قال الأَزهري:<br>\
النعائمُ منزلةٌ من منازل القمر، والعرب تسمّيها النَّعامَ الصادرَ، وهي<br>\
أَربعة كواكب مُربَّعة في طرف المَجَرَّة وهي شاميّة، ويقال لها<br>\
النَّعام؛ أَنشد ثعلب:<br>\
باضَ النَّعامُ به فنَفَّر أَهلَه،<br>\
إلا المُقِيمَ على الدّوَى المُتَأَفِّنِ<br>\
النَّعامُ ههنا: النَّعائمُ من النجوم، وقد ذكر مستوفى في ترجمة بيض.<br>\
ونُعاماكَ: بمعنى قُصاراكَ. وأَنْعَم أن يُحْسِنَ أَو يُسِيءَ: زاد.<br>\
وأَنْعَم فيه: بالَغ؛ قال:<br>\
سَمِين الضَّواحي لم تَُؤَرِّقْه، لَيْلةً،<br>\
وأَنْعَمَ، أبكارُ الهُمومِ وعُونُها<br>\
الضَّواحي: ما بدا من جَسدِه، لم تُؤرّقْه ليلةً أَبكارُ الهموم<br>\
وعُونُها، وأَنْعَمَ أي وزاد على هذه الصفة، وأَبكار الهموم: ما فجَأَك،<br>\
وعُونُها: ما كان هَمّاً بعدَ هَمّ، وحَرْبٌ عَوانٌ إذا كانت بعد حَرْب كانت<br>\
قبلها. وفَعَل كذا وأَنْعَمَ أي زاد. وفي حديث صلاة الظهر: فأَبردَ<br>\
بالظُّهْرِ وأَنْعَمَ أي أَطال الإبْرادَ وأَخَّر الصلاة؛ ومنه قولهم: أَنْعَمَ<br>\
النظرَ في الشيءِ إذا أَطالَ الفِكْرةَ فيه؛ وقوله:<br>\
فوَرَدَتْ والشمسُ لمَّا تُنْعِمِ<br>\
من ذلك أَيضاً أَي لم تُبالِغْ في الطلوع.<br>\
ونِعْمَ: ضدُّ بِئْسَ ولا تَعْمَل من الأَسماء إلا فيما فيه الألفُ<br>\
واللام أو ما أُضيف إلى ما فيه الأَلف واللام، وهو مع ذلك دالٌّ على معنى<br>\
الجنس. قال أَبو إسحق: إذا قلت نِعْمَ الرجلُ زيدٌ أو نِعْمَ رجلاً زيدٌ،<br>\
فقد قلتَ: استحقّ زيدٌ المدحَ الذي يكون في سائر جنسه، فلم يجُزْ إذا كانت<br>\
تَسْتَوْفي مَدْحَ الأَجْناسِ أن تعمل في غير لفظ جنسٍ. وحكى سيبويه:<br>\
أَن من العرب من يقول نَعْمَ الرجلُ في نِعْمَ، كان أصله نَعِم ثم خفَّف<br>\
بإسكان الكسرة على لغة بكر من وائل، ولا تدخل عند سيبويه إلا على ما فيه<br>\
الأَلف واللام مُظَهَراً أو مضمراً، كقولك نِعْم الرجل زيد فهذا هو<br>\
المُظهَر، ونِعْمَ رجلاً زيدٌ فهذا هو المضمر. وقال ثعلب حكايةً عن العرب: نِعْم<br>\
بزيدٍ رجلاً ونِعْمَ زيدٌ رجلاً، وحكى أَيضاً: مررْت بقومٍ نِعْم قوماً،<br>\
ونِعْمَ بهم قوماً، ونَعِمُوا قوماً، ولا يتصل بها الضمير عند سيبويه<br>\
أَعني أَنَّك لا تقول الزيدان نِعْما رجلين، ولا الزيدون نِعْموا رجالاً؛<br>\
قال الأزهري: إذا كان مع نِعْم وبِئْسَ اسمُ جنس بغير أَلف ولام فهو نصبٌ<br>\
أَبداً، وإن كانت فيه الأَلفُ واللامُ فهو رفعٌ أَبداً، وذلك قولك نِعْم<br>\
رجلاً زيدٌ ونِعْم الرجلُ زيدٌ، ونَصَبتَ رجلاً على التمييز، ولا تَعْملُ<br>\
نِعْم وبئْس في اسمٍ علمٍ، إنما تَعْمَلانِ في اسم منكورٍ دالٍّ على<br>\
جنس، أو اسم فيه أَلف ولامٌ تدلّ على جنس. الجوهري: نِعْم وبئس فِعْلان<br>\
ماضيان لا يتصرَّفان تصرُّفَ سائر الأَفعال لأَنهما استُعملا للحال بمعنى<br>\
الماضي، فنِعْم مدحٌ وبئسَ ذمٌّ، وفيهما أَربع لغات: نَعِمَ بفتح أَوله وكسر<br>\
ثانيه، ثم تقول: نِعِمَ فتُتْبع الكسرة الكسرةَ، ثم تطرح الكسرة الثانية<br>\
فتقول: نِعْمَ بكسر النون وسكون العين، ولك أَن تطرح الكسرة من الثاني<br>\
وتترك الأَوَّل مفتوحاً فتقول: نَعْم الرجلُ بفتح النون وسكون العين،<br>\
وتقول: نِعْمَ الرجلُ زيدٌ ونِعم المرأَةُ هندٌ، وإن شئت قلت: نِعْمتِ<br>\
المرأَةُ هند، فالرجل فاعلُ نِعْمَ، وزيدٌ يرتفع من وجهين: أَحدهما أَن يكون<br>\
مبتدأ قدِّم عليه خبرُه، والثاني أن يكون خبر مبتدإِ محذوفٍ، وذلك أَنَّك<br>\
لمّا قلت نِعْم الرجل، قيل لك: مَنْ هو؟ أو قدَّرت أَنه قيل لك ذلك فقلت:<br>\
هو زيد وحذفت هو على عادة العرب في حذف المبتدإ، والخبر إذا عرف المحذوف<br>\
هو زيد، وإذا قلت نِعْم رجلاً فقد أَضمرت في نِعْمَ الرجلَ بالأَلف<br>\
واللام مرفوعاً وفسّرته بقولك رجلاً، لأن فاعِلَ نِعْم وبِئْسَ لا يكون إلا<br>\
معرفة بالأَلف واللام أو ما يضاف إلى ما فيه الأَلف واللام، ويراد به تعريف<br>\
الجنس لا تعريفُ العهد، أو نكرةً منصوبة ولا يليها علَمٌ ولا غيره ولا<br>\
يتصل بهما الضميرُ، لا تقول نِعْمَ زيدٌ ولا الزيدون نِعْموا، وإن أَدخلت<br>\
على نِعْم ما قلت: نِعْمَّا يَعِظكم به، تجمع بين الساكنين، وإن شئت حركت<br>\
العين بالكسر، وإن شئت فتحت النون مع كسر العين، وتقول غَسَلْت غَسْلاً<br>\
نِعِمّا، تكتفي بما مع نِعْم عن صلته أي نِعْم ما غَسَلْته، وقالوا: إن<br>\
فعلتَ ذلك فَبِها ونِعْمَتْ بتاءٍ ساكنة في الوقف والوصل لأَنها تاء<br>\
تأْنيث، كأَنَّهم أَرادوا نِعْمَت الفَعْلةُ أو الخَصْلة. وفي الحديث: مَن<br>\
توضَّأَ يومَ الجمعة فبها ونِعْمَت، ومَن اغْتَسل فالغُسْل أَفضل؛ قال ابن<br>\
الأثير: أَي ونِعْمَت الفَعْلةُ والخَصْلةُ هي، فحذف المخصوص بالمدح،<br>\
والباء في فبها متعلقة بفعل مضمر أي فبهذه الخَصْلةِ أو الفَعْلة، يعني<br>\
الوضوءَ، يُنالُ الفضلُ، وقيل: هو راجع إلى السُّنَّة أي فبالسَّنَّة أَخَذ<br>\
فأَضمر ذلك. قال الجوهري: تاءُ نِعْمَت ثابتةٌ في الوقف؛ قال ذو الرمة:<br>\
أَو حُرَّة عَيْطَل ثَبْجاء مُجْفَرة<br>\
دَعائمَ الزَّوْرِ، نِعْمَت زَوْرَقُ البَلدِ<br>\
وقالوا: نَعِم القومُ، كقولك نِعْم القومُ؛ قال طرفة:<br>\
ما أَقَلَّتْ قَدَمايَ إنَّهُمُ<br>\
نَعِمَ السَّاعون في الأَمْرِ المُبِرّْ<br>\
هكذا أَنشدوه نَعِمَ، بفتم النون وكسر العين، جاؤوا به على الأَصل ولم<br>\
يكثر استعماله عليه، وقد روي نِعِمَ، بكسرتين على الإتباع. ودقَقْتُه<br>\
دَقّاً نِعِمّا أي نِعْمَ الدقُّ. قال الأَزهري: ودقَقْت دواءً فأَنْعَمْت<br>\
دَقَّه أي بالَغْت وزِدت. ويقال: ناعِمْ حَبْلَك وغيرهَ أَي أَحكمِه.<br>\
ويقال: إنه رجل نِعِمّا الرجلُ وإنه لَنَعِيمٌ.<br>\
وتَنَعَّمَه بالمكان: طلَبه. ويقال: أَتيتُ أَرضاً فتَنَعَّمَتْني أي<br>\
وافقتني وأَقمتُ بها. وتَنَعَّمَ: مَشَى حافياً، قيل: هو مشتق من النَّعامة<br>\
التي هي الطريق وليس بقويّ. وقال اللحياني: تَنَعَّمَ الرجلُ قدميه أي<br>\
ابتذَلَهما. وأَنْعَمَ القومَ ونَعَّمهم: أتاهم مُتَنَعِّماً على قدميه<br>\
حافياً على غير دابّة؛ قال:<br>\
تَنَعَّمها من بَعْدِ يومٍ وليلةٍ،<br>\
فأَصْبَحَ بَعْدَ الأُنْسِ وهو بَطِينُ<br>\
وأَنْعَمَ الرجلُ إذا شيَّع صَديقَه حافياً خطوات. وقوله تعالى: إن<br>\
تُبْدوا الصَّدَقاتِ فنِعِمَّا هي، ومثلُه: إنَّ الله نِعِمّا يَعِظكم به؛<br>\
قرأَ أَبو جعفر وشيبة ونافع وعاصم وأَبو عمرو فنِعْمّا، بكسر النون وجزم<br>\
العين وتشديد الميم، وقرأَ حمزة والكسائي فنَعِمّا، بفتح النون وكسر العين،<br>\
وذكر أَبو عبيدة <br>\
(* قوله «وذكر أَبو عبيدة» هكذا في الأصل بالتاء، وفي<br>\
التهذيب وزاده على البيضاوي أبو عبيد بدونها) حديث النبي، صلى عليه وسلم،<br>\
حين قال لعمرو بن العاص: نِعْمّا بالمالِ الصالح للرجل الصالِح، وأَنه<br>\
يختار هذه القراءة لأَجل هذه الرواية؛ قال ابن الأَثير: أَصله نِعْمَ ما<br>\
فأَدْغم وشدَّد، وما غيرُ موصوفةٍ ولا موصولةٍ كأَنه قال نِعْمَ شيئاً<br>\
المالُ، والباء زائدة مثل زيادتها في: كَفَى بالله حسِبياً حسِيباً ومنه<br>\
الحديث: نِعْمَ المالُ الصالحُ للرجل الصالِح؛ قال ابن الأثير: وفي نِعْمَ<br>\
لغاتٌ، أَشهرُها كسرُ النون وسكون العين، ثم فتح النون وكسر العين، ثم<br>\
كسرُهما؛ وقال الزجاج: النحويون لا يجيزون مع إدغام الميم تسكينَ العين<br>\
ويقولون إن هذه الرواية في نِعْمّا ليست بمضبوطة، وروي عن عاصم أَنه قرأَ<br>\
فنِعِمَّا، بكسر النون والعين، وأَما أَبو عمرو فكأَنَّ مذهَبه في هذا كسرةٌ<br>\
خفيفةٌ مُخْتَلَسة، والأصل في نِعْمَ نَعِمَ نِعِمَ ثلاث لغات، وما في<br>\
تأْويل الشيء في نِعِمّا، المعنى نِعْمَ الشيءُ؛ قال الأزهري: إذا قلت<br>\
نِعْمَ ما فَعل أو بئس ما فَعل، فالمعنى نِعْمَ شيئاً وبئس شيئاً فعَل،<br>\
وكذلك قوله: إنَّ اللهَ نِعِمّا يَعِظُكم به؛ معناه نِعْمَ شيئاً يَعِظكم<br>\
به.والنُّعْمان: الدم، ولذلك قيل للشَّقِر شَقائق النُّعْمان. وشقائقُ<br>\
النُّعْمانِ: نباتٌ أَحمرُ يُشبَّه بالدم. ونُعْمانُ بنُ المنذر: مَلكُ العرب<br>\
نُسب إِليه الشَّقيق لأَنه حَماه؛ قال أَبو عبيدة: إن العرب كانت<br>\
تُسَمِّي مُلوكَ الحيرة النُّعْمانَ لأَنه كان آخِرَهم. أَبو عمرو: من أَسماء<br>\
الروضةِ الناعِمةُ والواضِعةُ والناصِفةُ والغَلْباء واللَّفّاءُ.<br>\
الفراء: قالت الدُّبَيْرِيّة حُقْتُ المَشْرَبةَ ونَعَمْتُها <br>\
(* قوله<br>\
«ونعمتها» كذا بالأصل بالتخفيف، وفي الصاغاني بالتشديد) ومَصَلْتها <br>\
(* قوله<br>\
«ومصلتها» كذا بالأصل والتهذيب، ولعلها وصلتها كما يدل عليه قوله بعد<br>\
والمصول) أي كَنسْتها، وهي المِحْوَقةُ. والمِنْعَمُ والمِصْوَلُ:<br>\
المِكْنَسة.<br>\
وأُنَيْعِمُ والأُنَيْعِمُ وناعِمةُ ونَعْمانُ، كلها: مواضع؛ قال ابن<br>\
بري: وقول الراعي:<br>\
صبا صَبْوةً مَن لَجَّ وهو لَجُوجُ،<br>\
وزايَلَه بالأَنْعَمينِ حُدوجُ<br>\
الأَنْعَمين: اسم موضع. قال ابن سيده: والأَنْعمان موضعٌ؛ قال أَبو<br>\
ذؤيب، وأَنشد ما نسبه ابن بري إلى الراعي:<br>\
صبا صبوةً بَلْ لجَّ، وهو لجوجُ،<br>\
وزالت له بالأَنعمين حدوجُ<br>\
وهما نَعْمانانِ: نَعْمانُ الأَراكِ بمكة وهو نَعْمانُ الأَكبرُ وهو<br>\
وادي عرفة، ونَعْمانُ الغَرْقَد بالمدينة وهو نَعْمانُ الأَصغرُ. ونَعْمانُ:<br>\
اسم جبل بين مكة والطائف. وفي حديث ابن جبير: خلقَ اللهُ آدمَ مِن<br>\
دَحْنا ومَسحَ ظهرَ آدمَ، عليه السلام، بِنَعْمان السَّحابِ؛ نَعْمانُ: جبل<br>\
بقرب عرفة وأَضافه إلى السحاب لأَنه رَكَد فوقه لعُلُوِّه. ونَعْمانُ،<br>\
بالفتح: وادٍ في طريق الطائف يخرج إلى عرفات؛ قال عبد الله ابن نُمَير<br>\
الثَّقَفِيّ:<br>\
تضَوَّعَ مِسْكاً بَطْنُ نَعْمانَ، أنْ مَشَتْ<br>\
به زَيْنَبٌ في نِسْوةٍ عَطرات<br>\
ويقال له نَعْمانُ الأَراكِ؛ وقال خُلَيْد:<br>\
أَمَا والرَّاقِصاتِ بذاتِ عِرْقٍ،<br>\
