&lt;أَجج&gt;<br>\
: ( الأَجِيجُ : تَلَهُّبُ النّارِ ) .<br>\
ابنُ سِيدَه : الأَجَّةُ والأَجِيج : صوتُ النّار .<br>\

أَصرِفُ وَجهِي عن أَجِيجِ التَّنُّورْ<br>\

وأَجَّتِ النّارُ تَئِجُّ وتَؤُجُّ أَجِيجاً ، إِذا سَمِعْتَ صوتَ لَهَبِها ، قال :<br>\
كأَنَّ تَردُّدَ أَنفاسِه<br>\
أَجِيجُ ضِرَامٍ زَفَتْه الشَّمَالْ<br>\
( كالتَّأَجُّجِ ) والائْتِجاجِ .<br>\
( وأَجَّجْتُهَا تَأْجِيجاً ، فتَأَجَّجَتْ ، وائْتَجَّتْ ) ، على افْتَعَلَتْ .<br>\
وأَجِيجُ الكِيرِ : حَفيفُ النّارِ ، والفِعْل كالفِعْل ، وفي حديث الطُّفَيْل : ( طَرَفُ سَوْطِه يتَأَجَّجُ ) أَي يُضِىءُ ، من أَجِيجِ النّارِ : تَوَقُّدِها .<br>\
وفي الأَساس : أَجَّجَ النّارَ ، فأَجَّتْ وتَأَجَّجَتْ ، وهَجِيرٌ أُجَاج ، للشّمْسِ فيه مُجَاج .<br>\
( وأَجَّ الظَّلِيمُ يَئِجُّ ) ، بالكسر ، ( ويَؤُجُّ ) ، بالضَّمّ ، أَجًّا ، وأَجِيجاً والوجهانِ ذَكَرهما الصَّاغانِيّ في التكملة وابن منظور في اللّسان ، وعلى الضّمّ اقتصرَ الجَوْهَرِيّ والزّمخشَرِيّ ، وهو على غير قياس ، والكسرُ نقله الصّاغانيّ عن ابن دُرَيْدٍ ، وقد رَدّهَا عليه أَبو عمرو في فَائِتِ الجَمْهَرة ، قاله شيخُنا : ( عَدَا وله حَفِيفٌ ) ، وفي اللّسان : سُمِعَ حَفِيفهُ في عَدْوِه ، قال يَصِف ناقةً :<br>\
فَرَاحَتْ وأَطْرَافُ الصُّوَى مُحْزَئِلَّةٌ<br>\
تَئِجُّ كَما أَجَّ الظَّلِيمُ المُفَزَّعُ<br>\
وأَجَّ الرَّجُلُ يَئِجُّ أَجيجاً : صَوَّتَ ، حكاه أَبو زيد ، وأَنشد لجَمِيل :<br>\
تَئِجُّ أَجِيجَ الرَّحْلِ لمَّا تَحَسَّرَتْ<br>\
مناكِبُها وابْتُزَّ عَنْهَا شَلِيلُها<br>\
وأَجّ يَؤُجّ أَجًّا : َسْرَعَ ، قال :<br>\
سَدَا بِيَدَيْهِ ثُمّ أَجَّ بسَيْرِه<br>\
كأَجِّ الظَّلِيمِ من قَنِيصٍ وكالِبِ<br>\
وفي التَّهْذِيب : أَجَّ في سيره يَؤُجّأَجًّا ، إِذا أَسْرَعَ وهَرْوَلَ وأَنْشَد :<br>\
يَؤُجُّ كما أَجَّ الظَّلِيمُ المُنَفَّرُ<br>\
قال ابن بَرِّيّ : صوابُه تَوجُّ ، بالتّاءِ ، لأَنه يصف ناقَتَه ، ورواه ابن دُرَيْد : ( الظَّلِيم المُفَزّع ) .<br>\
وفي حديث خَيْبَر : ( فلما أَصْبَحَ دَعَا عَلِيًّا ، فأَعْطَاه الرَّايَةَ ، فخَرَجَ بها يَؤُجُّ حَتّى رَكَزَها تَحْتَ الحِصْنِ ) الأَجُّ : الإِسراع والهَرْوَلَةُ ، كما في النِّهاية .<br>\
وفي الأَساس : ومن المجاز : مَرَّ يَؤُجُّ في سَيرِه ، أَي له حَفِيفٌ كاللَّهَبِ ، وقد أَجَّ أَجَّةَ الظَّليمِ .<br>\
وسَمِعْتُ أَجَّتَهُم حَفِيفَ مَشْيِهم واضطِّرابِهِمْ .<br>\
( والأَجَّةُ : الاخْتِلاطُ ) ، وفي اللسان : أَجَّةُ القَوْمِ ، وأَجِيجُهُم : اخْتلاطُ<br>\
كلامِهم مع حَفِيفِ مَشْيِهِم ، وقولهم : القَوْمُ في أَجَّةٍ ، أَي في اخْتِلاطٍ .<br>\
( و ) الأَجَّةُ ، والائْتِجاجُ ، والأَجِيجُ والأُجاجُ : ( شِدّةُ الحَرِّ ) وتَوَهُّجُه ، والجمع إِجاجٌ ، مثل : جَفْنَةٍ وجِفَانٍ .<br>\
( وقد ائْتَجَّ النّهَارُ ) على افتعل ، ( وتَأَجَّ وتَأَجَّجَ ) .<br>\
ويقال : جاءَت أَجَّةُ الصيْفِ ، قال رؤبة :<br>\
وحَرَّقَ الحَرُّ أُجَاجاً شَاعِلاَ<br>\
وقال ذُو الرُّمَّة :<br>\
بِأَجَّةٍ نَشَّ عنها الماءُ والرُّطُبُ<br>\
( و ) يقال : ( ماءٌ أُجَاجٌ ) بالضّمّ ، أَي ( مِلْحٌ ) ، وقيل : ( مُرٌّ ) ، وقيل : شَديدُ المَرارةِ ، وقيلَ : الأُجاجُ : الشَّديدُ الحَرارةِ وكذالك الجَمْعُ ، قال الله عزّ وجلّ : { وَهَاذَا مِلْحٌ أُجَاجٌ } ( سورة الفرقان ، الآية : 53 ) وهو الشّديدُ المُلُوحَةِ والمَرَارَةِ ، مثل ماءِ البحر ، وفي حديث عليّ ( عَذْبُها أُجَاجٌ ) .<br>\
وهو الماءُ المِلْحُ الشديدُ المُلُوحَةِ ، كذا نُقلَ عن ابن عباس في تفسيرِ ، وفي حديث الأَحنف : ( نَزَلْنا سَبِخَةً نَشّاشَةً ، طَرَفٌ لها بالفَلاةِ وطَرَفٌ لها بالبَحْرِ الأُجَاجِ ) .<br>\
ونَقَل شيخُنَا عن بعضِ أَئمّة الاشتقاق الأُجَاجُ بالضّمّ ، من الأَجِيجِ ، وهو تَلَهُّبُ النّارِ ، فكلّ ما يَحْرِقُ الفَمَ مِن مالِحٍ ومُرَ أَو حارَ فهو أُجَاجٌ .<br>\
وعن الحسن : هو ما لا يُنْتَفَعُ به في شُرْبٍ أَو زَرْعٍ أَو غيرِهِما .<br>\
( وقد أَجَّ ) الماءُ يَؤُجَّ ( أُجُوجاً ، بالضّمّ ) في مصدره ومضارعه ، أَي فهو من باب كتَبَ ، ومثلُه في الصّحاح واللّسان ( وأَججْتُه ) ، بالتخفيف .<br>\
( وَيَأْجَجُ ، كيَسْمَعُ ) ، أَي بالفتح على القياس ، حكاه سيبويه ، ( ويَنْصُرُ<br>\
ويَضْرِبُ ) الأَخِير حكاه السّيرافيّ عن أَصحاب الحديث ، ونقله الفَرّاءُ عن المُفَصلِ ( : ع بمَكّة ) ، شَرّفها الله تعالى .<br>\
( واليَأْجُوجُ ) باللاّم مُشتقٌّ ( من ) أَجَّ ( يَئِجُّ هاكذا وهاكذا ) إِذا هَرْوَلَ وَعَدَا .<br>\
( ويَأْجُوجُ ومَأْجُوجُ ) : قَبِيلتانِ من خَلْقِ الله تعالى ، وجاءَ في الحديث : ( أَن الخَلْقَ عَشَرَةُ أَجزاءٍ ، تسعةٌ منها يَأْجُوجُ ومَأْجُوجُ ) وهما اسْمَان أَعْجَمِيَّان ، جاءَت القراءَةُ فيهما بهَمْزٍ وغير هَمْز ، و ( مَنْ لا يَهْمِزُهُمَا ) و ( يجعل الأَلِفَيْنِ زَائِدَتَيْن ) يقول : إِنّهُما ( من يَجَجَ ومَجَجَ ) ، وهما غير مصروفين ، قال رُؤْبَةُ :<br>\
لَوْ أَن يَأْجُوج ومَأْجُوجَ مَعَا<br>\
وعادَ عَادٍ واسْتَجَاشُوا تُبَّعَا<br>\
ومن هَمَزَهُمَا قال : إِنّهما من أَجَّتِ النَّارُ ، ومن الماءِ الأُجَاجِ ، وهو الشّديدُ المُلُوحَةِ المُحْرِقُ من مْلُوحَتهِ ، ويكون التَّقديرُ في يَأْجُوجَ يَفْعُول ، وفي مَأْجُوجَ مَفْعُول ، كأَنَّه من أَجيج النّارِ .<br>\
قالوا : ويجوز أَن يكون يَاجُوج فاعولا ، وكذالك ماجُوج ، وهذا لو كان الاسمان عَرَبِيَّيْن لكان هذا اشتقاقَهما فأَمّا الأَعْجَمِيّةُ فلا تُشْتَقُّ من العَرَبيّة .<br>\
( وقَرَأَ ) أَبو العَجَّاجِ ( رُؤْبَةُ ) بنُ العَجّاجِ ( : آجُوجَ ومَاجُوجَ ) بقلب الياءِ همزاً .<br>\
( و ) قرأَ ( أَبُو مُعَاذٍ : يَمْجُوجَ ) بقلب الأَلفِ الثّانيةِ ميماً .<br>\
( والأَجُوجُ ) كَصَبُورٍ : ( المُضِىءُ النَّيِّر ) ، عن أَبي عمرٍ و ، وأَنشد لأَبي ذُؤَيْب يعصف بَرْقاً :<br>\
يُضِىءُ سَناه رَاتِقاً مُتَكَشِّفاً<br>\
أَغَرَّ كمِصْباحِ اليَهُودِ أَجُوجُ<br>\
قال ابنُ بَرِّيّ : يصفُ سَحاباً مُتَتابعاً ، والهاءُ في سَناهُ تعود على السّحاب ، وذالك أَنّ البَرْقَةَ<br>\
إِذا بَرَقَت انكَشَفَ السّحابُ ، وراتِقاً : حالٌ من الهاءِ في سناه ، ورواه الأَصمعيّ ( راتق متكشفٌ ) بالرفع ، فجعل الرّاتِقَ البَرْقَ ، كذا في اللسان .<br>\
( وأَجَجَ ، كَمَنَع : حَمَلَ على العَدُوّ ) ، هاكذا في سائر النّسخ التي بأَيْدِينا ، وهو قولُ أَبي عمرٍ و ، وتَمامُه : وجَأَجَ إِذا وَقَفَ جُبْناً ، وأَنكر شيخُنَا ذالك ، وقال : أَيُّ مُوجِبٍ للفتح مع عدم حرف الحلق فيه ؟ وصَوّبَ التشديد ، ونَسِيَ القاعدة الصّرفية أَنّه لا يُشْتَرَطُ أَن اللفْظ إِذا كان من باب مَنَعَ لا بُد فيه من أَحَدِ حروف الحَلْقِ ، وإِنما إِذا وُجِدَ فيا اللّفْظِ أَحدُ حروفٍ الحَلْقِ ، أَي في عينه أَو لامه ، فإِنّه مفتوكٌ دائِماً ومع أَنّ الصاغانيّ هكذا ضَبَطَه بالتخْفِيف في تكْمِلَته .<br>\
[ ] ومما يستدرك عليه :<br>\
أَجَّجَ بينهم شَرًّا : أَوقَدَه ، وقول الشاعر :<br>\
تَكَفُّحَ السمائِمِ الأَوَاجِجِ<br>\
إِنّمَا أَراد الأَوَاجّ ، فاضطُرَّ ، ففكّ الإِدخامَ .<br>\
وأَجِيجُ الماءِ : صَوتُ انصِبَابِه .<br>\
&lt;أَذج&gt;<br>\
: ( أَذجَ بالمُعْجَمَةِ ) ، إِذا ( أَكْثَرَ من شُرْبِ الشرَابِ ) ، عن أَبي عَمْرٍ و ، ومثله في التّكْمِلَة .<br>\
( وأَيْذَجُ ، كأَحْمَد ) إِنما أَرادَ الوَزْنَ فقط من غير ملاحظة إِلى الزوائد والأَصلية ، وابلاّ فأَلف أَحمد زائدة بخلاف المَوْزُون فإِنها أَصليّة ( : د ، بِكِرِسْتَانَ ) .<br>\
&lt;أَذربج&gt;<br>\
: [ ] ومما يستدرك عليه :<br>\
أَذْرَبِيجَانُ ، وهاذا محَلُّه ، وهو موضعٌ أَعجمِيّ ، مُعَربٌ ، قال الشّمّاخ :<br>\
تَذَكَّرْتُها وَهْناً وقد حالَ دُونَها<br>\
قُرَى أَذْرَبِيجانَ المَسالِحُ والجَالُ<br>\
وجعله ابنُ جِنِّي مُركَّباً ، قال : هاذا اسمٌ فيه خمسةُ موانعَ من الصرْف ، وهي : التعْرِيف ، والتّأْنِيث ، والعُجْمَةُ ، والتَّرْكِيبُ ، والأَلف والنّون ، كذا في اللسان .<br>\
&lt;أَرج&gt;<br>\
: ( الأَرجُ ، محرّكةً : ) نَفْحَةُ الرِّيحِ الطَّيّبَة .<br>\
( و ) عن ابنِ سيدَه ( الأرِيجُ والأَرِيجَةُ ) : الرِّيحُ الطَّيِّبَةُ ، وجمعُها الأَرَائِجُ ، وأَنشد ابنُ الأَعرابيّ :<br>\
كأَن رِيحاً من خُزامَى عالِجِ<br>\
أَوْرِيحَ مِسْكٍ طَيِّبِ الأَرَائِجِ<br>\
والأَرَج والأَرِيجُ : ( تَوَهُّجُ رِيحِ الطِّيبِ ) .<br>\
( أَرِجَ ) الطِّيبُ ، ( كَفَرِحَ ) يَأْرَج أَرَجاً ، فهو أَرِجٌ : فاحَ ، قال أَبو ذُؤَيْبٍ :<br>\
كَأَنَّ عليها بالَةً لَطَمِيَّةً<br>\
لَهَا مِن خِلال الدَّأْيَتَيْنِ أَرِيجُ<br>\
( والتَّأْريجُ : الإِغْرَاءُ والتَّحْرِيشُ ) في الحَرْبِ قال العجَّاجُ :<br>\
إِنّا إِذا مُذْكِي الحُرُوبِ أَرَّجَا<br>\
وأَرَّجْتُ بين القوْمِ تَأْرِيجاً ، إِذا أَغْرَيْتَ بينَهم وهَيَّجْتَ ، مثل أَرَّشْت ( كالأَرْجِ ) ثلاثيًّا .<br>\
وأَرَّجْتُ الحَرْبَ ، إِذا أَثَرْتَها .<br>\
( و ) التَّأْرِيجُ ، والإِرَاجَةُ ( : شَيْءٌ م ) أَي معروف ( في الحِسَابِ ) وسيأْتي قريباً .<br>\
( والأَرَجانُ مُحَرَّكَةً : سَعْيُ المُغْرِي ) بالإِغْرَاءِ بين النّاسِ ، وقد أَرَّجَ بينهم .<br>\
( و ) أَرَّجَانُ ، ( كهَيَّبانَ ) ، أَي<br>\
بتشديد المثنّاة التَّحتِيَّة مع فتحها : موضعٌ حكاه الفارسيّ ، وأَنشد :<br>\
أَرادَ الله أَن يُخْزِي بُجَيْراً<br>\
فسَلَّطَنِي عليهِ بَأَرَّجَانِ<br>\
وقيل : هو ( د ، بفارِسَ ) ، وخَفَّفَه بعضُ متأَخِّري الشُّعراءِ ، فأَقْدَم على ذلك لعُجْمَتِه ، كذا في اللّسان .<br>\
قلت : التّخفيف ورَدَ في قولِ المُتَنَبِّي ، وقال شُرّاحه : إِنه ضَرورة ، ويَدلّ لذالك قول الجوهريّ : وربما جاءَ في الشّعر بتخفيفِ الرّاءِ .<br>\
ثم إِنه هل هو فَعَّلان من أَرَجَ ، كما صنع المصنّف ؟ أَو هو أَفْعال من رَجَنَ ؟ أَو هو لفظٌ أَعجميٌّ فلا تُعْرَف مادّته ؟ .<br>\
وصوَّبَ الخفاجهيّ في شفاءِ الغليل أَنّه فَعّلان ، لا أَفْعَلان ؛ لئلا تكون الفاءُ والعين حرفاً واحداً ، وهو قليلٌ ، نقله شيخُنا .<br>\
( والأَرّاجُ ) والمِئْرَجُ ، ككَتّانٍ ومِنْبَرٍ ( : الكَذّابُ ) ، والخَلاّطُ ، ( والمُغْرِى ) بين النّاسِ .<br>\
( والمُؤَرَّجُ ، كمُحَمَّدٍ : الأَسَدُ ) من أَرَّجْتُ بين القومِ تَأْرِيجاً ، إِذا أَغْرَيْتَ بينهم وهَيَّجْتَ قال أَبو سَيد : ( و ) منه سُمِّي المُؤَرِّج ( بالكسرِ أَبو فَيْدٍ ) بفتح الفاءِ وسكون الياءِ التّحتيّة وآخره دال مهملة ، هاكذا في نُسختنا على الصّواب ، وتصَحَّفَ على شيخنا ، فذكر في شرحه ، المُقَابَلِ عليه أَبو قَبيللآ ، وهو خَطَأٌ ( : عَمْرُو ابنُ الحارِثِ السَّدِوسِيُّ ) النَّحْوِيّ البَصْرِيّ ، أَحدُ أَئمّة اللّغةِ والأَدب .<br>\
وفي البُغْيَةِ للجلال : عَمْرُو بنُ مَنِيع ابن حُصَيْن السَّدُوسِيّ ، وفي شروح الشّواهد للرَّضِيّ : المُؤَرِّج ، كمُحَدِّثٍ السُّلَمِيّ : شاعِر إِسلاميّ من الدّولة الأُمويّة ، وفي الصّحاح عن أَبي سعيدٍ ، ومنه المُؤَرِّجُ الذُّهْلِيّ جَدُّ المُؤَرِّجِ الرَّاوِيَة ، سُمِّيَ ( لتَأْرِيجِه الحَرْبَ )<br>\
وتَأْرِيشِها ( بينَ بَكْرٍ وتَغْلِبَ ) ، وهما قبيلتانِ عظيمتانِ .<br>\
( و ) في التهذيب : ( الأَوَارِجَةُ : من كُتُبِ أَصحابِ الدّواوِينِ ) في الخَراجه ونحوِه ، ويقال : هاذا كتابُ التَّأرِيجِ ، وهو ( مُعَرّبُ آوَارِه ، أَي النّاقِ ؛ لأَنه يُنْقَلُ إِليها الأَنْجِيذَجُ ) بفتح فسكون فكسر فسكون التَّحْتِيّة وذال وجِيم ( الذي يُثْبَتُ فيه ما على كُلِّ إِنسانٍ ، ثُمّ يُنْقَلُ إِلى جَرِيدةِ الإِخْرَاجَاتِ ، وهي عِدَّةُ أَوارِجَاتٍ ) ، وقد بسط فيه المصنّف الكلامَ لاحتياج الأَمْر إِليه ، وهو الأَعْرفُ به .<br>\
[ ] ومما يستدرك عليه :<br>\
ما في النهاية في الحديث : ( لما جاءَ نَعِيُّ عُمَرَ ، رضي الله تعالى عنه ، إِلى المَدَائِنِ أَرِجَ النّاسُ ) أَي ضَجُّوا بالبكاءِ ، قال : وهو من أَرِجَ الطِّيبُ ، إِذا فاحَ .<br>\
وأَرَّجَ بالسَّبُعِ كهَرَّجَ : إِما أَن تكونَ لغة ، وإِما أَن تكون بدلاً .<br>\
وأَرَج الحَقَّ بالبَاطِلِ يَأْرِجُه أَرْجاً : خَلَطَه .<br>\
وأَرَّج النّارَ وأَرَّثَها : أَوْقَدَهَا ، مُشَدَّدٌ ، عن ابن الأَعرابيّ .<br>\
والأَيارِجة : دواءٌ ، وهو معَرّب .<br>\
&lt;أَزج&gt;<br>\
: ( الأَزَجُ ، مُحَرَّكَةً : ضَرْبٌ مِنَ الأَبْنِيَةِ ) وفي الصّحاحِ ، والمِصْبَاحِ ، واللِّسان الأَزَجُ : بَيْتٌ يُبْنَى طُولاً ، ويقال له بالفارسية : أوستان ، ( ج : آزُجٌ ) ، بضمّ الزّاي ، ( وآزَاجٌ ) ، قال الأَعشَى :<br>\
بَنَاه سُلَيْمَانُ بنُ دَاوُودَ حِقْبَةً<br>\
له آزُجٌ صُمٌّ وطَيٌّ مُوَثَّقُ<br>\
( وإِزَجَةٌ كَفِيَلَةٍ ) .<br>\
( وبابُ الأَزَجِ ) ، مُحَرَّكَةً : مَحَلَّةٌ كبيرةٌ ( ببغدادَ ) ، وقد نُسِب إليها جماعة من المُحَدِّثِين .<br>\
( وأَزَّجَهُ  تَأْزِيجاً : بناهُ وطَوَّلَهُ ) .<br>\
( و ) أَزَجَ ارَّجلُ ( كنَصَر ، وفَرِحَ أُزُوجاً ) ، بالضَّمّ مصدر الأَوّل ، والذي في اللسان وغيره .<br>\
: وأَزَجَ في مِشْيَتِهِ يَأْزِجُ ، أَي كيضرِب ، هاكذا ضبط بالقَلم ، أُزوجاً : ( أَسْرَعَ ) : قال :<br>\
فَزَجَّ رَبْدَاءَ جَوَاداً تَأْزِجُ<br>\
فسَقَطَتْ من خَلْفِهِنّ تَنْشِجُ<br>\
( و ) أَزَجَ ( عنى : تَثاقَلَ حينَ استَعَنْتُه ) .<br>\
وفي أُخرى : اسْتَغَثْتُه .<br>\
( و ) الأَزِجُ ( ككَتِفٍ : الأَشِرُ ) .<br>\
والأُزُوجُ : سُرعةُ الشدّ .<br>\
وفَرَسٌ أَزُوجٌ .<br>\
وأَزَجِ العُشْبُ ، إِذا طال .<br>\
&lt;إِسبرنج&gt;<br>\
: [ ] ومما يستدرك عليه :<br>\
ما ورد في الحديث ( مَن لَعبَ بالإِسْبِرَنْجِ والنَّرْدِ فقد غَمَسَ يَدَه في دَمِ خِنْزِيرٍ ) قال ابن الأَثِير في النهاية هو اسم الفَرس الذي في الشَّطْرَنْجِ واللفْظة فارسيّة معرّبة .<br>\
&lt;أَسج&gt;<br>\
: ( الأُسُجُ ، بضَمَّتَيْنِ ) هي : ( النُّوقُ السَّرِيعَاتُ ، وأَصله الوُسُجُ ) بالوَاو ، ولذا لم يذكره هنا الجوهريّ ولا ابن منظور ، وسيأْتي في وسج .<br>\
&lt;أَشج&gt;<br>\
: ( الأُشَّجُ ، كَزُمَّجٍ ) ، أَي على وِزَانِ سبُكَّرٍ ( : دَواءٌ كالكُنْدُرِ ) ، وهو أَكثرُ استعمالاً من الأُشَّقِ .<br>\
&lt;أَمج&gt;<br>\
: ( الأَمَجُ مُحَرَّكَةً : حَرٌّ ، وعَطَشٌ ) يقال : صَيْفٌ ، أَمَجٌ ( و ) هو ( الشَّدِيدُ الحَرِّ ) وقيل : الأَمَجُ : شِدَّةُ الحَرِّ والعَطَشِ والأَخْذِ بالنَّفَسِ .<br>\
وقال الأَصمعيّ : الأَمَجُ : تَوَهُّجُ الحَرِّ ، وأَنشدَ للعجّاج :<br>\
حتّى إِذا مَا الصَّيفُ كان أَمَجَا<br>\
وَفَرَغَا من رَعْيِ ما تَلَزَّجَا<br>\
( و ) في حديث ابنِ عباس رضي الله عنهما ( حَتّى إِذا كَانَ بالكَدِيدِ ما بينَ عُسْفَانَ وأَمَج ) هُو مُحَرّكة ( ع ) بينَ مَكَّةَ والمدينة شرّفهما الله تعالى ، فيه مَزارعُ ، وأَنشد أَبو العباس المُبرّد :<br>\
حُمَيْدُ الّذِي أَمَجٌ دَارُه<br>\
أَخُو الخَمْرِ ذُو الشَّيْبَةِ الأَصْلَعُ<br>\
( و ) أَمِجَ ( كفَرِحَ : عَطِشَ ) ، يقال أَمِجَتِ الإِبِلُ ، تَأْمَجُ أَمَجاً ، إِذا اشتَدّ بها حَرٌّ أَو عَطَشٌ .<br>\
( و ) عن أَبي عَمرٍ و : أَمَجَ ( كضَرَبَ ) إِذا ( سَارَ ) سَيْراً ( شَدِيداً ) .<br>\
&lt;أَنبج&gt;<br>\
: [ ] ومما يستدرك عليه هنا :<br>\
ذكر الأَنْبِجَانِيَّةَ ( في الحديثِ : ( ايتُوني بأَنْبِجَانِيَّةِ أَبيه جَهْمٍ ) قال ابنُ الأَثِيرِ : قيل هي منسوبَةُ إِلى مَنْبِجَ ، المدينة المعروفِة ، وقيل : إِلى موضعٍ اسمه أَنْبِجانُ ، وهو أَشْبَهُ : لأَن الأَوّل فيه تَعَسُّفٌ ، قال : والهمزة في هزائدةٌ ، وسيأْتي في نبج مستوفأ ، إِن شَاءَ الله تعالى .<br>\
&lt;أَوج&gt;<br>\
: ( الأَوْجُ : ضِدُّ الهُبُوطِ ) : وهو من اصطلاحات المُنَجِّمِين ، أَورده في التكملة ، وأَغفله ابنُ منظور ، كالجَوْهَرِيّ ، وغيرهما .<br>\
&lt;أَيج&gt;<br>\
: وذكر شيخُنا هنا الأَبجِيّ ، بالموحدة ونَقله عن المِصْبَاح ، وهو تَصْحِيفٌ عن الإِيجيّ ، بالمثنّاة بدل الموحّدة ، فاعْلَمْ .<br>\
( إِيجُ ، بالكسر : د ، بِفَارِسَ ) وققد نُسِبَ إِليهَا كِبَارُ المحَدِّثِين .<br>\
&lt;بأَج&gt;<br>\
: ( بَأَجَه ، كمَنَعَه : صَرَفَه ) .<br>\
( و ) بَأَجَ ( الرَّجُلُ : صاحَ ، كبَأجَ ) بالتشديد .<br>\
( و ) في الصّحاح قولهم : ( اجْعَلِ البَأْجاتِ  بأجا واحِدا ، أَي لَوْناً ) واحداً ( وضَرْباً ) واحداً .<br>\
وهو مُعَرَّب ، وأَصله بالفارسية : بَاهَا ، أَي أَلوانُ الأَطعمة ، وهمْزُه هو الفَصيحُ الذي اقْتَصَرع عَلَيْه ثَعْلَبٌ في الفصيح ، ( وقد لا يُهْمَزُ ) ، صرّح به الجوهَرِيّ ، وبعضُ شُرّاحِ الفَصيحِ .<br>\
