وقد رَاحَ الإِنْسَانُ إِلى الشَّيْءِ يَراحُ : إِذا نَشِطَ وسُرَّ به ، وكذالك ارْتَاح .<br>\
وأَنشد :<br>\
وزَعَمْتَ أَنّك لا تَرَاحُ إِلى النِّسَا<br>\
وسَمِعْتَ قِيلَ الكاشِحِ المُتَرَدِّدِ<br>\
( و ) الرَّاحُ : هي ( الأَكُفّ ) .<br>\
ويقال : بل الرَّاحَة : بَطْنُ الكَفِّ ، والكَفُّ : الرّاحةُ معَ الأَصابع : قاله شيخنا ، ( كالرّاحات .<br>\
و ) عن ابن شُميل : الرَّاحُ من ( الأَراضي المُسْتَوِيَةُ ) الّتي ( فيها ظُهُورٌ واسْتِواءٌ تُنْبِت كثيراً ) ، جَلْدَةٌ ، وفي أَماكنَ منها سُهُولٌ وجَراثيمُ ، وليسَتْ من السَّيْلِ في شَيْءٍ ولا اللوادِي .<br>\
( واحِدَتُهما رَاحةٌ ) .<br>\
( وَرَاحَةُ الكَلْبِ : نَبْتٌ ) ، على التَّشبيه .<br>\
( وذو الرَّاحَةِ : سَيْفُ المُخْتَارِ بنِ أَبي عُبيدٍ ) الثّقفيّ .<br>\
( والرَّاحَةُ : العِرْسُ ) ، لأَنها يُسْتَرَاح إِليها .<br>\
( و ) الرَّاحَةُ من البيت : ( السّاحةُ ، وطَيُّ الثَّوْب ) ، وفي الحديث عن جَعفرٍ : ( نَاوَلَ رَجُلاً ثَوْباً جديداً فقال : اطْوِهِ على راحَتِه ) أَي طَيِّه الأَوّلِ .<br>\
( و ) الرَّاحَةُ : ( ع قُرْبَ حَرَضَ ) ، وفي نسخة : و : ع ، باليمن وسيأْتي حَرُضُ .<br>\
( و ) الرَّاحَة : ( ع ببلادِ خُزَاعَةَ ، له يومٌ ) معروفٌ .<br>\
( وأَراحَ اللَّهُ العَبْدَ : أَدْخَلَه في الرَّاحَةِ ضِدّ التَّعب ) ، أَو في الرَّوْحِ وهو الرَّحمة ( و ) أَراح ( فلانٌ على فلانٍ : حَقَّه : رَدَّده عليه ) .<br>\
وفي نسخةٍ : ردّه .<br>\
قال الشاعر :<br>\
إِلاَّ -تُرِيحي علينا الحقَّ طائعةً<br>\
دونَ القُضاةِ فقاضِينا إِلى حكَمِ<br>\
وأَرِحْ عليه حَقَّه ، أَي رُدَّه .<br>\
وفي حديث الزُّبَيْر : ( لولا حُدودٌ فُرِضَتْ وفَرائضُ حُدَّتْ ، تُرَاحُ على أَهْلِها ) أَي تُرَدّ إِليهم ، والأَهلُ هم الأَئمَّة ؛ ويجوز بالعكس ، وهو أَنّ الأَئمَّةَ يَرُدُّونها إِلى أَهلِها من الرَّعِيّة .<br>\
ومنه حديث عائشةَ ( حتَّى أَراحَ الحَقَّ إِلى أَهْلِه ) ( كأَرْوَح .<br>\
و ) أَراحَ ( الإِبلَ ) وكذا الغنَمَ : ( رَدَّها إِلى المُراحِ ) وقد أَراحَها راعِيها يُرِيحُهَا ، وفي لغة : هَراحَها يُهْرِيحُها .<br>\
وفي حديث عُثْمَانَ رضي الله عنه : ( رَوَّحْتها بالعَشِيّ ) ، أَي رَدَدْتها إِلى المُزَاح .<br>\
وسَرَحتِ الماشيةُ بالغَداة ، وراحَتْ بالعَشِيّ ، أَي رَجَعتْ .<br>\
وفي ( المحكم ) : والإِراحةُ : رَدُّ الإِبلِ والغنَم من العَشِيِّ إِلى مُرَاحِهَا .<br>\
والمُرَاحُ : ( بالضم ) : المُنَاخُ ، ( أَي المَأْوَى ) حيث تَأْوِي إِليه الإِبلُ والغَنَمُ باللَّيْل .<br>\
وقال الفَيُّوميّ في المصباح عند ذِكْرِه المُرَاح بالضّم : وفتحُ الميمِ بهاذا المعنى خطأٌ ، لأَنه اسمُ مَكَانٍ ، واسمُ المكانِ والزّمانِ والمَصْدَرُ من أَفْعَل بالأَلف مُفْعَل بضمّ الميم على صيغة .<br>\
المفعول .<br>\
وأَمّا المَرَاحُ ، بالفتح : فاسمُ المَوْضِع ، من راحَتْ ،<br>\
بغير أَلفٍ ، واسمُ المكانِ من الثُّلاثيِّ بالفتح .<br>\
انتَهى .<br>\
وأَراحَ الرجلُ إِراحةً وإِراحاً ، إِذا راحَتْ عليه إِبلُه وغَنمُه ومالُه ، ولا يكون ذالك إِلاّ بعدَ الزَّوالِ .<br>\
وقول أَبي ذُؤيب :<br>\
كأَنَّ مَصاعِبَ زُبَّ الرُّؤُو<br>\
سِ في دارِ صِرْمٍ تَلاَقَى مُرِيحَا<br>\
يمكن أَن يكون أَراحَتْ ، لغة في راحَتْ ، ويكون فاعلاً في معنَى مَفْعول .<br>\
ويُروَى : ( تُلاقِي مُرِيحاً ) أَي الرَّجلَ الذي يُرِيحها .<br>\
( و ) أَراحَ ( الماءُ واللَّحْمُ : أَنْتَنَا ) ، كأَرْوَح .<br>\
يقال أَرْوَحَ اللَّحْمُ ، إِذا تَغيَّرَتْ رائحتُه ، وكذالك الماءُ .<br>\
وقال اللِّحْيَانيّ وغيرُه : أَخَذَتْ فيه الرِّيحُ وتَغَيَّرَ .<br>\
وفي حديث قَتَادَة : ( سُئلَ عن الماءِ الّذِي قد أَرْوَحَ : أَيُتَوَضَّأُ به قال : لا بَأْسَ ) أَرْوَحَ الماءُ وأَراحَ ، إِذا تَغيَّرتْ رِيحُه ؛ كذا في ( اللسان ) والغَرِيبَيْن .<br>\
( و ) أَراحَ ( فُلارنٌ : ماتَ ) ، كأَنَّه اسْتَرَاحَ .<br>\
وعبارةُ الأَساس : وتقول : أراحَ فأَراحَ ( أَي مات ) فاسْتُرِيحَ منه .<br>\
قال العجَّاج :<br>\
أَراحَ بعدَ الغَمَّ والتَّغَمْغُمُ وفي حديث الأَسْوَد بن يَزيدَ ( إِن الجَمَلَ الأَحمرَ لَيُرِيحُ فيه من الحَرّ ) الإِراحَةُ هنا : المَوْتُ والهَلاَكُ .<br>\
ويُرْوَى بالنون ، وقد تقدّم .<br>\
( و ) أَراح : ( تَنَفَّسَ ) .<br>\
قال امرؤُ القيس يَصف فَرساً بسَعَة المَنْخَرَيْن :<br>\
لها مَنْخَرٌ كَوِجَارِ السِّبَاعِ<br>\
فَمِنْه تُريحُ إِذا تَنْبَهِرْ<br>\
( و ) أَراحَ الرَّجُلُ : استراحَ و ( رَجَعتْ إِليه نَفْسُه بعدَ الإِعياءِ ) .<br>\
ومنه حديثُ أُمِّ أَيْمَنع ( أَنها عَطشَتْ مهاجِرةً في يومٍ شديدِ الحرِّ ، فدُلِّيَ إِليها دَلْوٌ من السَّماءِ ، فشَرِبتْ حتَّى أَراحَتْ ) وقال اللِّحْيانيّ : وكذالك<br>\
أَراحَت الدّابَّةُ .<br>\
وأَنشد :<br>\
تُرِيحُ بعد النَّفَسِ المَحفوز<br>\
( و ) أَراحَ الرجُلُ : ( دَخَلَ في الرِّيحِ ) .<br>\
ومثله رِيحَ ، مَبنيًّا للمفعول ، وقد تقدَّم .<br>\
( و ) أَراحَ ( الشَّيْءَ ) ورَاحَه يَرَاحُه ويَرِيحُه : إِذا ( وَجَدَ رِيحَه ) .<br>\
وأَنشد الجوهريّ بيت الهُذَليّ :<br>\
وماءٍ وَرَدْتُ على زَوْرةٍ<br>\
إِلخ ، وقد تقدّمَ .<br>\
وعبارة الأَساس : وأَرْوَحْت منه طِيباً : وَجدتُ رِيحَه .<br>\
قلت : وهو قول أَبي زيدٍ .<br>\
ومثله : أَنْشَيْت منه نَشْوَة ، ورِحْتُ رائحةً طَيِّبَةً أَو خَبيثَةً ، أَرَاحُهَا وأَرِيحُها .<br>\
وأَرَحْتها وأَرْوَحْتُهَا : وَجَدْتها .<br>\
( و ) أَراحَ ( الصَّيْدُ ) : إِذا ( وَجَدَ رِيحَ الإِنْسِيِّ ، كأَرْوَحَ ) في كلّ مما تقدَّم .<br>\
وفي ( التهذيب ) : وأَرْوَحَ الصَّيْدُ واسْتَرْوَحَ واسْتَرَاحَ : إِذا وَجَدَ رِيحَ الإِنسانِ .<br>\
قال أَبو زيد : -أَرْوَحَنِي الصَّيْدُ والضَّبُّ إِرْوَاحاً ، وأَنْشَأَنِي إِنْشَاءً ، إِذا وَجَدَ رِيحَك ونَشْوَتَك .<br>\
( وتَروَّحَ النَّبْتُ ) والشَّجرُ : ( طالَ ) .<br>\
وفي ( الرَّوْض الأُنُف ) : تَروَّحَ الغُصْنُ : نَبَتَ وَرَقُه بعد سُقوطِه .<br>\
وفي ( اللسان ) : تَرَوُّحُ الشَّجَرِ : خُرُوجُ وَرَقِه إِذا أَوْرَقَ النَّبْتُ في استقبالِ الشتاءِ .<br>\
( و ) تَروَّحَ ( الماءُ ) ، إِذا ( أَخَذَ رِيحَ غَيْرِه ، لقُرْبِه ) منه .<br>\
ومثله في ( الصّحاح ) .<br>\
ففي أَرْوَحَ الماءُ وتَرَوَّحَ نَوْعٌ من الفرق .<br>\
وتَعَقَّبَه الفيّوميّ في المصباح ، وأَقرّه شيخنا ، وهو محلُّ تأَمُّلٍ .<br>\
( وتَرْوِيحَةُ شَهْرِ رَمَضَانَ ) : مَرَّةٌ واحدةٌ من الرَّاحَة ، تَفْعيلةٌ منها ، مثل تَسليمة من السَّلام .<br>\
وفي ( المصباح )<br>\
أَرِحْنا بالصَّلاة : أَي أَقِمْها ، فيكون فِعْلُهَا راحةً ، لأَنّ انتظارها مَشقَّة على ( النفس ) وصلاَةُ التَّرَاوِيح مُشتقَّةٌ من ذالك ، ( سُمِّيَتْ بها لاسْتِرَاحةِ ) القَوْمِ ( بعدَ كلِّ أَرْبَعِ رَكَعَاتٍ ) ، أَو لأَنهم كانوا يَسترِيحون بين كلّ تَسليمتينِ .<br>\
( واسْتَرْوَحَ ) الرَّجلُ : ( وَجَدَ الرّاحَةَ ) .<br>\
والرَّوَاحُ والرَّاحَةُ : من الاستراحةِ .<br>\
وقد أَراحَني ، ورَوَّح عني ، فاسْترَحْتُ .<br>\
وأَرْوَحَ السَّبُعُ الرِّيحَ وأَرَاحَها ( كاسْتَراحَ ) واسْتَرْوَحَ : وَجَدَها .<br>\
قال اللِّحْيَانيّ : وقال بعضُهم : رَاحَها ، بغير أَلف ، وهي قليلة .<br>\
واسْتَرْوحَ الفَحْلُ واسْتَرَاحَ : وَجَدَ رِيحَ الأُنْثَى .<br>\
( و ) أَرْوَحَ الصَّيْدُ واسْتَرْوَحَ واستراحَ ، وأَنشأَ : ( تَشَمَّمَ ، و ) استرْوَحَ ، كما في ( الصّحاح ) ، وفي غيره من الأُمهات : استراحَ ( إِليه : اسْتَنامَ ) ، ونقلَ شيخُنَا عن بعضهم : ويُعدَّى بإِلى لتَضمُّنه معنَى يَطمئِنّ ويَسْكُن ، واستعمالُه صحيحاً شذُوذٌ .<br>\
انتهَى .<br>\
والمُسْتراحُ : المَخْرجُ .<br>\
( والارْتِيَاح : النَّشَاطُ ) .<br>\
وارتاحَ للأَمرِ : كرَاحَ .<br>\
( و ) الارتياحُ : ( الرَّحْمَةُ ) والرَّاحةُ .<br>\
( وارْتاحَ اللَّهُ له برحْمَته : أَنْقَذَه من البَلِيَّة ) .<br>\
والّذي في ( التهذيب ) : ونَزَلَتْ به بَلِيّةٌ فارْتَاحَ اللَّهُ له برحمتِه فأَنْقَذه منها .<br>\
قال رُؤْبة :<br>\
فارتاحَ رعبِّي وأَرَاد رَحْمَتِي<br>\
ونِعْمَةً أَتَمَّها فتَمَّتِ<br>\
أَراد : فارتاحَ : نَظَرَ إِليّ ورَحمَنِي .<br>\
قال : وقولُ رُئبةَ في فِعْلِ الخَالِق قالَه بأَعْرَابِيْتِه .<br>\
قال : ونحن نَسْتَوْحِشُ مِن مثل هاذا اللَّفْظِ ، لأَنّ الله تعالى إِنما يُوصَف بما وَصَف به نَفْسَه ، ولولاَ أَنّ الله تعالى هَدانا بفضله لتَمْجِيده وحَمْدِه بصفاته الّتي أَنْزَلها في كِتَابه ، ما كنّا لنهتديَ لها أَو نَجْتَرِىءَ عليها .<br>\
قال ابن سيده : فأَمّا الفارِسيّ فجعلَ هاذا البيتَ من جَفَاءِ الأَعْرَاب .<br>\
( والمُرْتَاح ) بالضّمّ : ( الخَامِسُ من خَيْلِ الحَلْبَةِ ) والسِّبَاقِ ، وهي عَشَرةٌ ، وقد تقدّمَ بعضُ ذكرِهَا .<br>\
( و ) المُرْتاح : ( فَرَسُ قيسِ الجُيُوشِ الجَدَلِيّ ) ، إِلى جَدِيلَةَ بنتِ سُبَيعٍ ، من حِمْيَر ، نُسِب وَلَدُهَا إِليها .<br>\
( والمُرَاوَحَة بين العمَلينِ : أَن يَعْمَلَ هاذا مَرَّةً وهاذا مَرَّةً ) .<br>\
وهما يَتَراوَحَانِ عَمَلاً ، أَي يَتَعَاقَبانه .<br>\
ويَرْتَوِحَان مِثْلُه .<br>\
قال لَبيد :<br>\
وولَّى عامِداً لِطِيَاتِ فَلْجٍ<br>\
يُرَاوِحُ بين صَوْنٍ وابْتِذَالِ<br>\
يعني يَبْتَذِل عَدْوَه مَرَّةً ويصون أُخْرَى ، أَي يَكُفّ بعد اجتهادٍ .<br>\
( و ) المُراوَحَةُ ( بين الرِّجْلينِ أَن يَقوم على كلِّ ) واحدةٍ منهما ( مَرَّةً ) .<br>\
وفي الحديث : أَنّه كان يُرَاوِحُ بين قَدمَيْه من طُولِ القِيَامِ ، أَي يَعتمد على إِحداهُمَا مرّةً ، وعلى الأَخرى مرَّةً ، ليوصِلَ الرَّاحةَ إِلى كلَ منهما .<br>\
ومنه حديثُ ابنِ مسعودٍ أَنه أَبصرَ رجلاً صَافًّا قَدَمَيْه فقال : ( لورَاوَحَ كانَ أَفْضَلَ ) .<br>\
( و ) المُرَاوَحَة ( بين جَنْبَيْه : أَن يَتَقَلَّب من جَنْب إِلى جَنْب ) .<br>\
أَنشد يعقوب :<br>\
إِذا أَجْلَخَذَّ لم يَكَدْ يُرَاوِحُ هِلْبَاجةٌ حَفَيْسَأٌ دُحادِحُ ( و ) من المجاز عن الأَصمعيّ : يقال ( رَاحَ للمَعْرُوفِ يَرَاحُ رَاحةً : أَخَذَتْه له خِفَّةٌ وأَرْيَحِيَّة ) ، وهي الهَشَّةُ .<br>\
قال الفارِسيّ : ياءُ أَرْيَحِيَّة بَدَلق من الواوِ .<br>\
وفي ( اللسان ) : يقال : رِحْتُ للمعروف أَراح رَيْحاً وارْتَحْتُ ارْتياحاً : إِذا مِلْتُ إِليه وأَحببْته .<br>\
ومنه قولهم :<br>\
أَرْيَحِيٌّ : إِذا كان سَخِيًّا يَرتاحُ للنَّدَى .<br>\
( و ) من المجاز : راحتْ ( يَدُه لكذا : خَفَّت ) .<br>\
وراحت يده بالسَّيْف ، أَي خَفَّتْ إِلى الضَّرْب به .<br>\
قال أُمَيَّةُ بنُ أَبي عائِذٍ الهُذَليّ :<br>\
تَرَاحُ يَدَاهُ بِمَحْشورةٍ<br>\
خَوَاظِي القِداحِ عجَافِ النِّصالِ<br>\
أَراد بالمحشورة ، نَبْلاً ، للُطْفِ قَدِّها ، لأَنه أَسْرَعُ لها في الرَّمْيِ عن القَوْس .<br>\
( ومنه ) ، أَي من الرَّواح بمعنَى الخِفَّةِ ( قولُه صلّى الله ) تعالى ( عليه وسلُّم ) : ( مَنْ راحَ إِلى الجمعة في السّاعةِ الأُولَى فكَأَنّما قَدَّمَ بَدَنَةً ، ( ومَنْ راحَ في الساعةِ الثانيةِ ) الحديث ) ، أَي إِلى آخرِه ( لم يُرِد رَوَاحَ ) آخِرِ ( النَّهَارِ ، بل المُرَاد خَفَّ إِليها ) ومَضَى .<br>\
يقال : راحَ القَوْمُ وتَرَوَّحوا ، إِذا سارُوا أَيَّ وَقْتٍ كَانَ .<br>\
وقيل : أَصْلُ الرَّوَاحِ أَن يكون بعد الزَّوال .<br>\
فلا تكون الساعاتُ الّتي عَدَّدَهَا في الحديث إِلاّ في ساعَةٍ واحدةٍ من يوم الجُمُعَةِ ، وهي بعد الزَّوَال ، كقولك : قَعَدْتُ عندَك ساعةً ، إِنما تُريد جزْءًا من الزَّمَان ، وإِن لم يكن ساعَةً حقيقة الّتي هي جزءٌ من أَربعةٍ وعشرين جُزْءًا مجموع اللَّيْلِ والنَّهَار .<br>\
( و ) راح ( الفَرَسُ ) يَارحُ راحَةً : إِذا تَحَصَّنَ ، أَي ( صار حِصَاناً ، أَي فَحْلاً ) .<br>\
( و ) من المجاز : راح ( الشَّجَرُ ) يَارحُ ، إِذا ( تَفطَّرَ بالوَرَق ) قبل الشِّتَاءِ من غير مَطَرٍ .<br>\
وقال الأَصمعيّ : وذالك حين يَرْدُ اللَّيلُ فيتَفطَّر بالوَرَقِ من غير معطرٍ .<br>\
وقيل : رَوَّحَ الشَّجرُ إِذَا تفطَّرَ بوَرَقٍ بعد إِدْبَارِ الصَّيْفِ .<br>\
قال الرَّاعِي :<br>\
وخالَفَ المجْدَ أَقوامٌ لهم ورِقٌ<br>\
رَاحَ العِضَاهُ به والعِرْقُ مَدْخُولُ<br>\
ورواه أَبو عَمْرٍ و : وخادَعَ الحَمْدَ أَقْوَامٌ ، أَي تَرَكوا الحَمْدَ ، أَي لَيْسُوا من أَهلِه .<br>\
وهاذه هي الرِّوَايَة الصُّحيحةُ .<br>\
( و ) راح ( الشَّيْءَ يَرَاحُه ويَرِيحُه ) : إِذا ( وَجَدَ رِيحَه ، كأَراحَه وأَرْوَحَه ) .<br>\
وفي الحديث : ( من أَعانَ على مُؤمِن أَو قَتَلَ مُؤمناً لم يُرِحْ رائحةَ الجَنّة ) ، من أَرَحْت ، ( ولم يَرِحْ رائحة الجنة ) من رِحْت أَراحُ .<br>\
قال أَبو عمرٍ و : هو من رِحْتُ الشَّيءَ أَريحُه : إِذا وجَدْت رِيحَه .<br>\
وقال الكسائيّ : إِنما هو لم يُرِح رائحةَ الجنّةِ ، من أَرَحْتُ الشَّيْءَ فأَنا أُرَيحه ، إِذا وَجَدّت رِيحَه ؛ والمعنى واحدٌ .<br>\
وقال الأَصمعيّ : لا أَدري : هو من رِحْتُ أَو أَرَحْت .<br>\
( و ) راحَ ( منك مَعْرُوفاً : نالِ ، كأَراحَه ) .<br>\
( والمَرْوَحَة ، كمَرْحَمة : المَفازَة ، و ) هي ( المَوْضِع ) الذي ( تَخْتَرِقه الرِّياحُ ) وتَتعاوَرُه .<br>\
قال :<br>\
كأَنَّ راكِبَها غُصْنٌ بمَرْوَحَةٍ<br>\
إِذا تَدَلّتْ به أَو شارِبٌ تَمِلُ<br>\
والجمع المَراوِيح .<br>\
قال ابن بَرّيّ : البيت لعُمَرَ بن الخطَّاب رَضِيَ اللَّهُ عنهُ .<br>\
وقيل : إِنه تمَثَّل به ، وهو لغيره ، قاله وقد ركب راحِلته في بعضِ المَفَاوِزِ فأَسرعَتْ ، يقول : كأَنّ راكبَ هاذه النَّاقَةِ لسُرْعتِهَا غُصْنٌ بمَوْضعٍ تَعْتَرِق فيه الرِّيحُ ، كالغُصْنِ لا يَزَالُ يَتمَايَلُ يميناً وشمالاً ، فشَبَّهَ راكبَها بغُصْنَ هاذه حالُه أَو شاربٍ ثَمِلٍ يَتمايلُ من شِدَّةِ سُكْرِه .<br>\
قلْت : وقد وَجدْت في هامش ( الصّحاح ) لابن القَطّاع قال : وجدْت أَبا محمدٍ الأَسود الغَنجانيّ قد ذَكَر أَنه لم يُعْرَف قائلُ هاذا البيتِ .<br>\
قال : وقرأْت في شعر عبد الرَّحمانِ بن حَسّان قصيدةً ميميّة :<br>\
كأَنَّ راكِبَها غُصْنٌ بمرْوَحةٍ<br>\
لدْنُ المجسَّةِ لَيْنُ العُودِ من سَلَمِ<br>\
لا أَدري أَهو ذاك فغُيِّر أَم لا ، وفي الغَرِيبينِ للهروِيّ أَنّ ابن عُمر ركِبَ ناقةً فارهةً فمَسَتْ به مشْياً جيّداً ، فقال : كأَنّ صاحِبها .<br>\
الخ .<br>\
وذكر أَبو زكرِيّا في تهذيب الإِصلاح أَنه بيت قديم تمثل به عمر بن الخطابِ رَضِيَ اللَّهُ عنه .<br>\
( و ) المِرْوَحة ، بكسر الميم ، ( كمِكْنَة ، و ) قال اللِّحْيَانيّ : هي المِرْوَحُ مثل ( مِنْبَرٍ ) : وإِنما كُسِرت لأَنها ( آلة يُتروَّحُ بها ) .<br>\
والجمع المَرَاوِحُ .<br>\
ورَوَّحَ عليه بها .<br>\
وتَروَّحَ بنفْسِه .<br>\
وقطع بالمرْوَحَة مَهَبّ الرِّيح .<br>\
وفي الحديث : ( فقد رأَيتُهم يتَرَوَّحُونَ في الضُّحى ) ، أَي احتاجُوا إِلى التَّرْوِيح من الحَرّ بالمرْوَحَة ، أَو يكون من الرَّوَاح : العَوْدِ إِلى بُيُوتِهم ، أَو من طَلبِ الرَّاحَة .<br>\
( والرائحة : النَّسيمُ طَيِّباً ) كان ( أَونَتْناً ) بكسر المثناة الفوقيَّة وسُكُونها .<br>\
وفي ( اللّسان ) : الرّائحة : رِيحٌ طَيِّبةٌ تَجِدُهَا في النَّسيم ، تقول : لهاذه البَقْلَةِ رائحةٌ طَيِّبةٌ .<br>\
ووجَدْتُ ريحَ الشيْءِ ورائحته ، بمعنًى .<br>\
( والرَّوَاحُ والرَّواحَةُ والرّاحَة والمُرَايحة ) ، بالضَّمّ ، ( والرَّوِيحَة ، كسَفِينَةٍ : وِجْدانُك ) الفَرْجَة بعد الكُرْبَة .<br>\
والرَّوْحُ أَيضاً : السُّرُورُ والفَرَحُ .<br>\
واستعارَه عليٌّ رَضِيَ اللَّهُ عنهُ لليَقِينِ فقال : ( باشرُوا رَوْحَ اليَقِينِ ) ، قال ابن سيده : وعندي أَنه أَراد ( السُّرورَ الحادِثَ من اليَقِينِ ) .<br>\
( وراحَ لذالك الأَمرِ يَراحُ رَوَاحاً ) كسَحَاب ( ورُؤُوحاً ) ، بالضّمّ ، ( ورَاحاً ورِيَاحَةً ) ، بالكسر ، وأَرْيَحِيّة ( : أَشْرَفَ له وفَرِحَ ) به ، وأَخذَتْه له خِفَّةٌ وأَرْيَحِيَّة .<br>\
قال الشاعر :<br>\
إِنّ البَخِيلَ إِذَا سأَلتَ بَهَرْتَهُ<br>\
وتَرَى الكَريمَ يَرَاحُ كالمُخْتال<br>\
وقد يُستَعَار للكِلاب وغيرِهَا .<br>\
أَنشد اللِّحْيَانيّ :<br>\
خُوصٌ تَراحُ إِلى الصِّيَاح إِذَا غَدَتْ<br>\
فِعْل الضِّرَاءِ تَرَاحُ لِلْكَلاّبِ<br>\
وقال اللّيث : راح الإِنسانُ إِلى الشّيءِ يَراح : إِذَا نَشِطَ وسُرَّ به ، وكذالك ارتاح .<br>\
وأَنشد :<br>\
وزَعَمْتَ أَنك لا تَراحُ إِلى النِّسَا<br>\
وسَمِعْتَ قِيلَ الكَاشِحِ المُترَدِّدِ<br>\
والرِّيَاحَة : أَن يَراحَ الإِنسانُ إِلى الشيْءِ فيَسْتَرْوِحَ وَيَنْشَطَ إِليه .<br>\
( والرَّوَاحُ ) : نَقِيضُ الصَّبَاح ، وهو اسْمٌ للوقْتِ .<br>\
وقيل : الرَّوَاحُ : ( العَشِيّ ، أَو من الزَّوال ) ، أَي من لَدُن زَوَالِ الشَّمْس ( إِلى اللَّيْل ) .<br>\
يقال : رَاحُوا يَفْعَلُون كذا وكذا ، ( ورُحْنَا رَوَاحاً ) ، بالفَتْح ، يعني السَّيْرَ بالعَشِيِّ وسار القَوْمُ رَوَاحاً ، وراحَ القَوْمُ كذالك ، ( وتَروَّحْنَا : سِرْنَا فيه ) ، أَي في ذالك الوَقتَ ، ( أَو عَمِلْنَا ) .<br>\
أَنشد ثعلب :<br>\
وأَنتَ الّذي خُبِّرتُ أَنّك رَاحلٌ<br>\
غَدَاةَ غَدٍ أَو رائحٌ بِهَجيرِ<br>\
والرَّوَاحُ قد يكون مصدر قَوْلِك : راح يَرُوح رَوَاحاً ، وهو نَقِيضُ قولِك : غَدا يَغْدُو غُدُوًّا .<br>\
( و ) تقول : ( خَرَجُوا برِيَاح من العشِيّ ) ، بكسر الرَّاءِ ، كذا هو في نُسخةِ ( التَّهْذيب ) و ( اللِّسَان ) ، ( ورَوَاحٍ ) ، بالفتح ، ( وأَرْوَاحٍ ) ، بالجمع ، ( أَي بأَوَّلٍ ) .<br>\
وقولُ الشاعر :<br>\
ولقدْ رَأَيْتُك بالقَوَادِمِ نَظْرةً<br>\
وَعَلَيَّ مِنْ سَدَفِ العَشِيّ رِيَاحُ<br>\
بكسر الراءِ فَسَّره ثعلب فقال : معناه وَقْتٌ .<br>\
ورَاحَ فلانٌ يَرُوح رَوَاحاً : من ذَهَابِه أَو سَيْرِه بالعَشَيّ .<br>\
قال الأَزهريّ : وسَمعْت العَرَبَ تستعملُ الرَّوَاحَ في السَّيْرِ كلَّ وَقْتٍ ، تقول : رَاح القَوْمُد إِذا سارُوا وغَدَوْا .<br>\
( ورُحْتُ القَوْمَ ) رَوْحاً ، ( و ) رُحْت<br>\
( إِليهم ، و ) رُحْتُ ( عندَهم رَوْحاً ورَوَاحاً ) ، أَي ( ذهبت إِليهم رَوَاحاً ) ، وراحَ أَهْلَه ، ( كرَوَّحْتُهم ) تَرْويحاً ( وتَروَّحْتُهم ) : جِئتُهم رَوَاحاً .<br>\
ويقول أَحدُهم لِصاحِبِه : تَرَّوحْ ، ويخاطِبُ أَصحابه فيقول : تَروَّحوا ، أَي سِيرُوا .<br>\
( والرَّوَائِحُ : أَمْطَارُ العِشِيِّ ، الواحِدَةُ رائحةٌ ) ، هشذه عن اللِّحيانيّ .<br>\
وقال مرَّةً : أَصابَتْنَا رائحةٌ ، أَي سَماءٌ .<br>\
( والرَّيِّحَة ، ككَيِّسة ، و ) الرِّيحَة ، مثل ( حِيلَة ) ، حكاه كُراع : ( النَّبْتُ يَظْهَرُ في أُصولِ العِضَاهِ الْتي بَقِيَتْ من عامِ أَوَّلَ ، أَو ما نَبَت إِذا مَسَّه البَرْدُ من غيرِ مطرٍ ) .<br>\
وفي ( التهذيب ) : الرَّيِّحة : نَبَاتٌ يَخْضَرّ بعدما يَبِس وَرقُه وأَعالي أَغصانِه .<br>\
وتَرَوَّحَ الشَّجَرُ وراحَ يَراحُ : تَفطَّرَ بالورق قبلَ الشِّتاءِ من غيرِ مَطرٍ .<br>\
وقال الأَصمعيّ : وذالك حين يَبْرُدُ اللَّيْلُ فيَتَفطَّرُ بالوعرقِ من غير مَطرٍ .<br>\
( و ) من المجاز : ( ما في وَجْهِه رائحةٌ ، أَي دَمٌ ) ، هاذه العبارةُ مَحلُّ تَأَمُّلٍ ، وهاكذا هي في سائر النُّسخِ الموجودَة والّذي نُقِل عن أَبي عُبيدٍ : يقال : أَتانَا فُلانٌ وما في وَجْهِه رائحةُ دَمٍ من الفَرَقِ ، وما في وجْهِه رائحةُ دمٍ أَي شَيْءٌ .<br>\
وفي ( الأَساس ) : وما في وَجْهِه رائحةُ دَمٍ ، إِذا جاءَ فَرِقاً .<br>\
فليُنْظَر .<br>\
( و ) من الأَمثال الدائرة : ( تَرَكْتُهُ على أَنْقَى من الرَّاحةِ ) ، أَي الكَفّ أَو السَّاحة ، ( أَي بلا شَيْءٍ ) .<br>\
( والرَّوْحاءُ ) ممدوداً ( : ع بين الحَرَميْنِ ) الشَّريفينِ ، زادَهما اللَّهُ شَرَفاً وقال عِيَاضٌ : إِنّه من عَمَلِ الفُرْعِ ، وقد رِدَّ ذالك ( على ثَلاثينَ أَو أَربعينَ ) أَو ستَّةٍ وثلاثينَ ( مِيلاً من المدينة ) ، الايخيرُ من كِتَاب مُسْلِمٍ .<br>\
قال شيخُنا : والأَقوالُ مُتَقَارِبَة .<br>\
وفي ( اللّسان ) : والنِّسبة إِليه : رَوْحَانِيٌّ ، على غيرِ قِياس .<br>\
( و ) الرَّوْحَاءُ ( : ة من رَحَبَةِ الشامِ ) ، هي رَحَبَةُ مالِكه بنِ طَوْق .<br>\
( و ) الرَّوْحاءُ ( : ة ) أُخرَى ( من ) أَعمال نَهرِ ( عِيسَى ) بنِ عَليِّ بنِ عبدِ الله بنِ عبَّاسٍ ، وهي كُورَةٌ واسِعَةٌ غَربِيَّ بَغدادَ .<br>\
( وعبدُ الله بنُ رَوَاحَةَ ) بنِ ثَعْلَبَة<br>\
الأَنصاريّ ، من بني الحارث بن الخَزْرَج ، أَبو محمد : ( صحابيّ ) نَقيبٌ ، بَدْريّ ، أَميرٌ .<br>\
( وبنو رَوَاحَةَ ) ، بالفتح : ( بَطْنٌ ) ، وهم بنو رَوَاحَةَ بنِ مُنْقِذِ بنِ عَمْرِو بن مَعِيصِ بن عامرِ بن لُؤَيّ بن غَالب بن فهْر ، وكان قد رَبعَ في الجَاهِلِيَّة ، أَي رَأَسَ على قَوْمِه ، وأَخذ المِرْباعَ .<br>\
( وأَبو رُويْحةَ ) الخَثْعَميّ ( كجُهَينةَ : أَخو بِلالٍ الحَبَشِيّ ) ، بالمُؤاخاة ، نَزَلَ دمشقَ .<br>\
( ورَوْحٌ اسمُ ) جَمَاعةٍ من الصّحابة والتَّابِعين ومَنْ بَعْدَهم ، منهم :<br>\
رَوْحُ بنْ حَبِيبٍ الثَّعْلَبيّ ، روي عن الصِّدِّيق ، وشَهِدَ الجابِيَةَ ؛ ذكرَه ابنُ فهدِ في ( مُعجم الصَّحابة ) .<br>\
ورَوْحُ بن سَيَّارٍ أَو سيَّارُ بنْ رَوْحٍ ، يقال : له صحبةٌ ؛ ذكرَه ابن مَنْدَه وأَبو نُعَيم .<br>\
ومنهم أَبو زُرْعَةَ رَوْحُ بن زِنْباعٍ الجُذاميّ ، من أَهلِ فلَسْطِينَ ، وكان مُجَاهِداً غازِياً ، رَوَى عنه أَهلُ الشّام ، يُعَدّ في التّابعين ، على الأَصحّ .<br>\
وَرَوْحُ بن يَزيدَ بن بَشيرٍ .<br>\

سمِعَ جَدَّه رِيَاحاً : أَنه حَجَّ مع عُمَرَ حَجَّتَيْن ؛ كذا في ( تاريخ البخاريّ ) .<br>\
( و ) رِيَاح ( بن عُبَيْدة ) هاكذا ، والصّواب : رِيَاح بن عُبَيْد ( الباهِليّ ) مولاهم ، بصريّ ، ويقال : كُوفيّ ، ويقال : بصريّ ، ويقال : كُوفيّ ، ويقال : حِجَازيّ ، والدُ مُوسى والخيَارِ .<br>\
( و ) رِيَاحُ ( بن عُبَيْدة ) السُّلَميّ ( الكُوفيّ ) عن ابن عُمَر ، وأَبي سَعيد الخُدْريّ ، وهما ( مُعَاصِرانِ لثَابتٍ البُنَانيّ ) الرّاوي عن أَنَسٍ .<br>\
( و ) رِياحُ ( بنْ يرْبُوعِ ) بنِ حَنْظَلَةَ بنِ مالكِ بن زَيْد مَناةَ بن تميم : ( أَبو القَبِيلةِ ) من تَميم ، منهم مَعْقِلُ بنُ قَيْس -الرِّياحيّ أَحدُ أَبْطَالِ الكُوفة وشُجْعانها .<br>\
( و ) رِيَاحُ بن عبد الله بن قُرْطِ بن رِزَاحِ بن عَدِيّ بن كَعب ( جدٌّ ) رابِعٌ ( لعُمَرَ بنِ الخَطَّابِ رَضِيَ اللَّهُ تعالى عنهُ ) وهو أَبو أَداة وعبد العُزَّى .<br>\
( و ) رِيَاحُ بن عَدِيّ الأَسْلَميّ ( جدٌّ لبُريدَةَ بنِ الحُصَيبِ ) بنِ عبد الله بن الحارث بن الأَعرج .<br>\
( و ) رِياحٌ ( جَدُّ لجَرْهَدِ ) بنِ خُوَيلد ، وقيل : ابن رِزَاح ( الأَسْلَميّ ) .<br>\
( ومُسْلِم بن رِيَاحٍ ) الثَّقفيّ ( صحابيّ ) ، رَوَى عنه عَوْنُ بن أَبي جُحيفةَ ، وقيل : رَبَاح ، بنقطة واحدة .<br>\
( و ) مُسلم بن رِيَاح ( تابعيٌّ ) مَوْلَى عليَ ، حدَّثَ عن الحُسين بن عليّ .<br>\
( وإِسماعيلُ بن رِيَاح ) بن عُبَيْدَةَ ، روَى عن جَدِّه المذكور أَوَّلاً ؛ كذا في كتاب ( الثِّقَات ) لابن حِبّان .<br>\
( وعُبيدةُ بن رِيَاح ) القِتْبانيّ عن مُثَبِّت ، وعنه ابنه الحارث .<br>\
( وعَبِيدُ بن رِيَاحٍ ) عن خَلاّد بن يَحيَى ، وعنه ابنُ أَبي حاتِم .<br>\
( وعُمَرُ بن أَبي عُمَرَ رِيَاحٍ ) أَبو حَفْص البَصريّ ، عن عَمْرِو بن شُعَيْبٍ وابنِ طاوُوس .<br>\

( والخِيَارُ وموسى ابنَا رِيَاحِ ) بن عُبيد الباهليّ البصريّ ، حَدَّثَا .<br>\
( وأَبو رِيَاحٍ مَنصورُ بنُ عبد الحميدِ ) وقيل : أَبو رَجاءٍ ، عن شُعْبَةَ ، ( مُحَدِّثونَ ) .<br>\
( واختُلِفَ في رِيَاحِ بن الرَّبيعِ ) الأُسَيِّديّ ( الصّحابيّ ) أَخي حَنْظَلَةَ الكاتبِ ، مَدَنيُّ ، نَزَلَ البَصْرَة ، روَى عنه حفيدُه المُرقِّع بن صَيْفِيّ ، وعنه قَيْسُ بن زُهَيْرٍ ، قال الدَّارَقُطْنيّ : رِيَاحٌ فَرْدٌ في الصَّحَابة .<br>\
وقال البخاريّ في ( التّاريخ ) : وقال بعضُهم : رِيَاحٌ ، يعني بالتّحتية ، ولم يثْبت .<br>\
