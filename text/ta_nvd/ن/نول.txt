[ (النَّهَلُ، مُحَرَّكَةً: أَوَّلُ الشَّرْبِ) ، وَالثَّانِي العَلَلُ، وَقد (نَهِلَتِ الإِبِلُ، كَفَرِحَ، نَهَلاً) ، مُحَركةً (وَمَنْهَلاً) ، مَصْدَرٌ مِيمِيٌّ، أَي: شَرِبَتْ فِي أَوَّل الوِرْدِ، وَمِنْهُ قَولُ الشّاعِر:<br>\
(وَقَدْ نَهِلَتْ مِنَّا الرِّماحُ وَعَلَّتِ ... )<br>\
(وَإِبِلٌ نَواهِلُ وَنِهالٌ) ، بالكَسْرِ، (وَنَهَلٌ، مُحَرّكَةً، وَنُهُولٌ) ، بالضَّمّ، (وَنَهَلةٌ) ، بالتَّحْرِيك، وَفِي بَعْضِ النُّسَخ: كَفَرِحَةٍ، (و) يُقالُ: إِبِلٌ (نَهْلَى) وَعَلَّى: لِلَّتِي تَشْرَبُ النَّهَلَ والعَلَلَ، قَالَ عاهانُ بنُ كَعْبٍ:<br>\
(تَبُكَّ الحَوْضَ عَلاَّها وَنَهْلَى ... وَدُونَ ذِيادِها عَطَنٌ مُنِيمُ)<br>\
وَقَدْ مَرَّ الكَلاَمُ عَلَيْهِ فِي &quot; ع ل ل &quot;.<br>\
(وَقَد أَنْهَلَها) : سَقاها أَوَّلَ الوِرْد، قَالَ:<br>\
(أَعَلَلاً وَنَحْنُ مُنْهِلُونَهْ ... )<br>\
(والمَنْهَلُ: المَشْرَبُ) ، وَمِنْهُ حَدِيثُ الدَّجّالِ: &quot; أَنَّهُ يَرِدُ كُلَّ مَنْهَلٍ &quot;. (و) قَالَ ثَعْلَبٌ: المَنْهَلُ: (الشُّرْبُ) ، قَالَ ابنُ سِيْدَه: وَهذا يَتَّجِهُ أَنْ يَكُونَ مَصْدَرَ نَهَلَ، وَقَدْ كَانَ يَنْبَغي أَنْ لاَ يَذْكُرَهُ، لِأَنَّهُ مُطَّرِد.<br>\
(و) أَيْضًا (المَوْضِعُ الَّذِي فِيْهِ المَشْرَبُ) ، عَن ثَعْلَب، (و) كَثُرَ ذلِكَ حَتَّى سُمِّيَ (المَنْزِلُ) الَّذي (يَكُونُ) لِلْسُّفّارِ (بالمَفازَةِ) مَنْهَلاً. وَقَالَ أَبو مالكٍ: المَنازِلُ والمَنَاهِلُ واحدٌ، وَهِي المَنازِلُ عَلى المَاءِ. وَقَالَ خَالِدُ بنُ جَنَبَةَ: المَنْهَلُ: كُلُّ مَا يَطَؤُهُ الطَّرِيقُ،<br>\
وَكُلُّ مَا كانَ عَلى غَيْرِ الطَّرِيقِ لاَ يُدْعَى مَنْهَلاً، وَلَكِن يُضافُ إِلى مَوْضِعِهِ، أَو إِلى مَنْ هُوَ مُخْتَصٌّ بِهِ، فَيُقَالُ: مَنْهَلُ بَنِي فُلاَنٍ؛ أَي: مَشْرَبُهُم وَمَوْضِعُ نَهَلِهِم. وَفي الصِّحَاح: المَنْهَلُ: عَيْنُ ماءٍ تَرِدُهُ الإِبِلُ فِي المَراعي، وَتُسَمَّى المَنازِلُ الَّتي فِي المَفاوِزِ عَلى طَريقِ السُّفّارِ مَناهِلَ؛ لِأَنَّ فِيها مَاء.<br>\
(والناهِلَةُ: المُخْتَلِفَةُ إِلى المَنْهَلِ) ، وَكَذلِكَ النازِلَةُ، قَالَ:<br>\
(وَلَمْ تُراقِبْ هُناكَ ناهِلَةَ الواشِينَ ... لَمَّا اجْرَهَدَّ ناهِلُها)<br>\
(وَأَنْهَلُوا: نَهِلَتْ إِبِلُهُمْ) ، أَيْ: شَرِبَتِ الوِرْدَ الأَوَّلَ فَرَوِيَتْ.<br>\
(والنَّهَلُ، مُحَرَّكَةً، مِنَ الطَّعامِ: مَا أُكِلَ) ، وَقَد وَرَدَ فِي كَلاَمِ بَعْضِهِم: أَكَلَ مِنَ الطَّعامِ حَتَّى نَهِلَ. قَالَ شَيْخُنَا: والظّاهِر أَنَّهُ مِنَ المَجاز، وَعلاَقَتُهُ لُزُومُ الشُّرْبِ لِلْأَكْلِ غَالِبًا، وَإِلاَّ فَالنَّهَلُ إِنَّما هُوَ فِي الشَّرابِ كالعَلَلِ.<br>\
(وَأَنْهَلَهُ: أَغْضَبَهُ) ، كَما فِي المُحْكَم. (والمِنْهالُ: الرَّجُلُ الكَثِيرُ الإِنْهال) لإِبْله.<br>\
(و) أَيْضًا (الكَثِيبُ العالِي) الَّذي (لاَ يَتَماسَكُ انْهِيارًا) عَن مَوْضِعه.<br>\
(و) قَالَ الفَرّاءُ: المِنْهالُ: (القَبْرُ، و) أَيْضًا: (الغايَةُ فِي السَّخاءِ، كالمِنْهَلِ فيهمَا) .<br>\
(و) المِنْهالُ: (أَرْضٌ) . (وَمِنْهالٌ القَيْسِيُّ، أَو صَوابُهُ مِلْحانٌ: صحابيّ) ، وَهوَ مِنْهالُ بنُ أَوْسٍ أَبو عَبْدِ المَلِك، لَهُ حَدِيثٌ فِي مُسْنَد أَحْمَد، هكَذا ذَكَره الذَّهَبِيُّ، وَقَالَ فِي مِلْحانَ مَا نَصُّه: مِلْحانُ بنُ شِبْلٍ البِكْرِيُّ وَقِيل القَيْسِيُّ وَالِدُ عَبْدِ المَلِكِ، لَهُ فِي صَوْم أَيّامِ البِيضِ فِي سُنَنِ أَبي داوُدَ.<br>\
(و) نُهَيْلٌ، (كَزُبَيْرٍ: اسمٌ) . (والنَّهْلاَنُ: الشَّارِبُ) ، عَن ابْنِ دُرَيْدٍ. (و) النَّهْلاَنُ: (الرَّيّانُ والعَطْشانُ، كالناهِلِ فيهمَا، كِلاَهُما ضِدٌّ) . وَفِي الصِّحاح قَالَ أَبو زَيْدٍ:<br>\
النّاهِلُ العَطْشانُ، والنّاهِلُ الرَّيّانُ، وَهُوَ مِنَ الأَضْدادِ، وَقَالَ النّابِغَةُ:<br>\
(الطّاعِنُ الطَّعْنَةَ يَوْمَ الوَغَى ... يَنْهَلُ مِنْها الأَسَلُ النّاهِلُ)<br>\
جَعَلَ الرِّماحَ كَأَنَّها تَعْطَشُ إِلى الدَّمِ فَإِذا شَرَعَتْ فِيهِ رَوِيَتْ. وَقَالَ أَبُو عُبَيْدٍ: هُوَ ههُنا الشّارِبُ، وَإِنْ شِئْتَ العَطَشانَ، أَي: يَرْوَى مِنْهُ العَطْشانُ. وَقَالَ أَبو الوَلِيد: يَنْهَلُ أَي: يَشْرَبُ مِنْهُ الأَسَلُ الشّارِبُ. وَقَالَ الأَزْهَرِيُّ: وَقَوْلُ جَرِيرٍ يَدُلُّ عَلى أَنَّ العِطاشَ تُسَمَّى نِهالاً:<br>\
(وأَخُوهُما السَّفّاحُ ظَمَّأَ خَيْلَهُ ... حَتَّى وَرَدْنَ جِبَا الكُلاَبِ نِهالاَ)<br>\
قَالَ: وَقَالَ عَمْرَة بن طارِقٍ فِي مِثْلِهِ:<br>\
(فَما ذُقْتُ طَعْمَ النَّوْمِ حَتَّى رَأَيْتُنِي ... أُعارِضُهُم وِرْدَ الخِماسِ النَّواهِلِ)<br>\
وَفِي حَدِيثِ لَقِيط: &quot; أَلاَ فَيَطَّلِعُونَ عَلَى حَوْضِ الرَّسُولِ لاَ يَظْمَأُ وَاللهِ ناهِلُه &quot;، يَقُولُ: مَنْ رَوِيَ مِنْهُ لَمْ يَعْطَشْ بَعْدَ ذلِكَ أَبَدًا. وَقَالَ شَيْخُنا: قَالَ جَماعَةٌ: إِنَّ تَسْمِيَةَ العَطْشانِ ناهِلاً إِنَّما هُو عَلى جِهَةِ التَّفاؤُلِ، كَالمَفَازَة.<br>\
(و) المُنْهِلُ، (كَمُحْسِنٍ: ماءٌ لِسُلَيْمٍ) . (والنَّواهِلُ: الإِبِلُ الجِياع) . (وانْهَلْ تَلاَنَ) ، كَذا فِي النُّسَخ، وَفِي العُبابِ: فُلاَن؛ (أَي: حَسْبُكَ الآنَ) ، عَنِ الفَرّاء. [] وَمِمّا يُسْتَدْرَكُ عَلَيْه:<br>\
النَّهَلُ: الرِّيُّ. والنَّهَلُ: العَطَشُ ضِدٌّ، والفِعْلُ كَالفعل، وَقَول كَعْبٍ:<br>\
(كَأَنَّهُ مُنْهَلٌ بِالرّاحِ مَعْلُولُ ... )<br>\
أَيْ: مَسْقِيٌّ بِالرَّاحِ، يُقَالُ: أَنْهَلْتُهُ فَهُوَ مُنْهَلٌ، وَفي حَديثِ معاوِيَةَ: &quot; النَّهَلُ الشُّرُوعُ &quot;، هُوَ جَمْعُ ناهِلٍ وَشَارعٍ، أَيْ: الإِبِلُ العِطَاشُ الشّارِعَةُ فِي المَاءِ، وَكَذلِكَ النَّوَاهِلُ. وَيُقَالُ: مِنْ أَيْنَ نَهِلْتَ اليَوْمَ، أَي: شَرِبْتَ فَرَوِيْتَ. وَقَوْلُهُ:<br>\
(مَا زَالَ مِنْهَا ناهِلٌ ونائبُ ... )<br>\
النّاهِلُ الَّذي رَوِيَ فَاعْتَزَلَ، وَالنّائِبُ الَّذِي يَنُوبُ عَوْدًا بَعْدَ شُرْبِهَا لِأَنَّهَا لَمْ تُنْضَحُ رِيًّا، وَقَالَ أَبو الهَيْثَم: نَاهِلٌ وَنَهَلٌ مِثْلُ خَادِمٍ وَخَدَمٍ، وَحارِسٍ وحَرَسٍ، وَجَمْعُ النَّهَلِ نِهَالٌ، كَجَبٍ ل وَجِبالٍ، قَالَ الرّاجِزُ:<br>\
(إِنَّكَ لَنْ تُثَأْثِئَ النِّهالاَ ... )<br>\
(بِمِثْلِ أَنْ تُدَارِكَ السِّجَالاَ ... )<br>\
واسْتَعَملَ بَعْضُ الأَغْفالِ النَّهَلَ فِي الدُّعَاءُ، فَقَالَ:<br>\
(ثُمَّ انْثَنَى مِنْ بَعْدِ ذَا فَصَلَّى ... )<br>\
(عَلَى النَّبِيِّ نَهَلاً وَعَلاَّ ... )<br>\
وَمِنْهَالُ بنُ عِصْمَةَ: رَجُلٌ مِنْ بَنِي يَرْبُوعَ، وَإِيّاهُ عَنَى مُتَمِّمُ بنُ نُوَيْرَةَ اليَرْبُوعِيُّ - رَضِيَ اللهُ تَعَالَى عَنْهُ -:<br>\
(لَقَدْ كَفَّنَ المِنْهَالُ تَحْتَ رِدَائِهِ ... فَتًى غَيْرَ مِبْطانِ العَشِيّاتِ أَرْوَعَا)<br>\
وَمِنْهالُ بن خَلِيفَةَ، وَمِنْهَالُ بن عَمْرٍ والأَسَدِيُّ: مُحَدِّثان. وَمِنَ المَجاز: أَسَدٌ نَاهِلٌ وَنَهّالٌ. وَأَنْهَلُوا دُرُوعَهُم: سَقَوْها السَّقْيَةَ الأُوْلَى.<br>\
&lt;نهـبل&gt;<br>\
(نَهْبَلَ) الرَّجُلُ، أَهْمَلَهُ الجَوْهَرِيُّ، وَقَالَ ابْنُ الأَعْرَابِيِّ: أَيْ (أَسَنَّ) . وَقَالَ اللَّيْثُ: (شَيْخٌ نَهْبَلٌ وَعَجُوزٌ نَهْبَلَةٌ) ، قَالَ أَبُو زُبَيْدٍ:<br>\
(مَأْوَى اليَتِيمِ وَمَأْوَى كُلِّ نَهْبَلَةٍ ... تَأْوِي إِلَى نَهْبَلٍ كالنَّسْر عُلْفُوفِ)<br>\
(والنَّهْبَلَةُ: مِشْيَةٌ فِي ثِقَلٍ) ، كالهَنْبَلَةِ، عَنِ ابْنِ دُرَيْدٍ، وَقَالَ ابْنُ الأَعْرَابِيِّ: هَنْبَلَ الرَّجُلُ: ظَلَعَ، وَمَشَى مِشْيَةَ الضَّبُعِ العَرْجاءِ، وَكَذلِكَ نَهْبَلَ.<br>\
(و) النَّهْبَلَةُ: (الناقَةُ الضَّخْمَةُ) ، قَالَ صَخْرُ بنُ عُمَيْرٍ:<br>\
(أَبْقَى الزَّمَانُ مِنْكِ نَابًا نَهْبَلَهْ ... )<br>\
(وَرَحِمًا عِنْدَ اللِّقَاحِ مُقْفَلَهْ ... )<br>\
(وَفِي) سُنَنِ (التِّرْمِذِيّ فِي حَدِيثِ الدَّجّالِ: &quot; فَيَطْرَحُهُم بالنَّهْبَلِ &quot;، وَهُوَ تَصْحِيفٌ، والصّوابُ) : بِالمَهْبِلِ، كَمَنْزِلٍ، (بِالمِيمِ) ، وَسَيَأْتِي فِي &quot; هـ ب ل &quot;.<br>\
&lt;نهـشل&gt;<br>\
(النَّهْشَلُ، كَجَعْفَرٍ: الذِّئْبُ، و) أَيْضًا: (الصَّقْرُ، وَاسْمُ) رَجُلٍ فِي العُبابِ، وَهُوَ نَهْشَلُ بنُ حَرِيٍّ: شَاعِرٌ، قَالَ سِيْبَويه: هُوَ يَنْصَرِفُ لِأَنَّهُ فَعْلَلٌ، وَإِذا كانَ فِي الكَلاَمِ مِثْلُ جَعْفَرٍ لَمْ يُمْكِن الحُكْمُ بِزِيادَةِ النُّونِ، كَما فِي الصِّحاحِ. قُلْتُ: وَإِلَيْهِ ذَهَب الجمْهُورُ، وَنَقَلَ الأَزْهَرِيُّ عَنِ الأَصْمَعِيِّ أَنَّهُ مُشْتَقٌّ مِنَ النَّهْشَلَةِ، وَهِيَ الكِبَرُ والاضْطِرابُ، وَذَهَبَ ابْنُ القَطَّاعِ إِلى زِيادَةِ لاَمِهِ، وَكَأَنَّهُ أَخَذَهُ مِنَ النَّهْشِ.<br>\
(و) نَهْشَلٌ: (قَبِيلَةٌ) مِنَ العَرَبِ، وَهُوَ نَهْشَلُ بنُ دارِمِ بنِ مالِكِ بنِ حَنْظَلَةَ بنِ مالِكِ بنِ زَيْدِ مَناةَ بنِ تَمِيمٍ، قَالَ الأَخْطَلُ:<br>\
(خَلاَ أَنَّ حَيًّا مِنْ قُرَيْشٍ تَفاضَلُوا ... عَلى الْنّاسِ أَوْ أَنَّ الأَكَارِمَ نَهْشَلاَ)<br>\
(و) النَهْشَلُ: (المُسِنُّ المُضْطَرِبُ كِبَرًا، أَو) الَّذِي أَسَنَّ (وفِيهِ بَقِيَّةٌ، وَهِيَ بِهاء) . (وَأَبُو نَهْشَلٍ: لَقِيطُ بنُ زُرَارَةَ التَّمِيمِيُّ) ، نَقَلَهُ الجَوْهَرِيُّ.<br>\
