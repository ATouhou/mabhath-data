( و ) الضَّبُّ : ( الغَيْظُ والحِقْدُ ) الكامِنُ في الصَّدْرِ كَذَا في الفرق لابن السّيد ، وقيل : هُوَ الضغن والعَدَاوَةُ .<br>\
( ويكسر ) ، وجمعه ضِبَابٌ .<br>\
قال الشاعر :<br>\
فَمَا زَالَتْ رُقَاكَ تَسُلُّ ضِغْنِي<br>\
وتُخْرِجُ مِن مَكامِنِها ضِبَابِي<br>\
وذكره الزَّمَخْشَرِيّ في الأَسَاس في<br>\
بَابِ المَجَازِ .<br>\
وقال آخر :<br>\
ولاتَكُ ذَا وَجْهَيْن يُبْدِي بَشَاشَةً<br>\
وفي قَلْبِه ضَبٌّ من الغِلِّ كَامِنُ<br>\
ورجل خَبٌّ ضَبٌّ : مُنْكَرٌ مُرَاوِغٌ حَرِبٌ .<br>\
وتَقُولُ : أَضَبَّ فلان على غِلَ في قَلْبِه أَي أَضْمَرَه .<br>\
وفي حديث عَليَ رضي الله عنه : ( كُلٌّ مِنْهُمَا حَامِلُ ضَبَ لصَاحِبِه ) .<br>\
وفي الأَسَاسِ ، مِنَ المَجَازِ : ورجل خَبٌّ ضَبٌّ يُشْبِه الضَّبَّ في خِدْعَتِه .<br>\
يقال : أَخْدَعُ مِنْ ضَبَ .<br>\
وامْرأَة خَبَّةٌ ضَبَّةٌ .<br>\
قُلْتُ : هذا المَثَلُ في حَياةِ الحَيَوَانِ والمُسْتَقْصَى .<br>\
( و ) الضَّبُّ : ( داءٌ ) يَأْخُذُ ( في الشَّفَةِ ) فَتَرِ وتَجْسُ وتَسِيلُ دَماً ويقال : تَجَسًى بمَعْنَى تَيَبَّس وتَصَلَّب .<br>\
( وقد ضَبَّتِ ) الشَّفَةُ ( تَضِبُّ ) بالكَسْرِ ( ضَبًّا وضُبُوباً .<br>\
و ) أَصْلُ الضَّبِّ : ( اللُّصُوقُ بالأَرْضِ ) ضَبّ ( يَضِب بالكَسْرِ في الكُلِّ ) .<br>\
قال شَيْخُنَا : وذكر الكسر مُسْتَدْرَك ، فإِنَّ إِتْبَاعَ المَاضِي بالمُضَارِع نَصٌّ في الكَسْرِ .<br>\
( والضَّبَّةُ ) والضَّبُّ : ( الطَّلْعَةُ قَبْلَ أَن تَنْفَلِق ) عن الغَرِيضِ .<br>\
والجَمْع ضِبَابٌ .<br>\
قال :<br>\
يُطِفْنَ بِفُحَالِ كأَنَّ ضِبَابَهُ<br>\
بُطُونُ المَوَالِي يَوْمَ عِيدٍ تَعَدَّت<br>\
يقول : طَلْعُهَا ضَخْمٌ كأَنَّه بُطُونُ مَوَالِ تَغَدَّوْا فتَضَلَّعُوا .<br>\
( و ) الضَّبَّةُ : ( مَسْكُ ) بالفَتْح ( الضَّبِّ يُدْبَغُ للسَّمْنِ ) أَي ليُجْعَل فيه .<br>\
( و ) الضَّبَّةُ : ( حَدِيدَة عَرِيضَة يُضَبَّبُ بِهَا ) البابُ والخَشَب .<br>\
والجَمْعُ ضِبَابٌ .<br>\
يقال : ضَبَبْتُ الخَشَبَ<br>\
ونَحْوَهُ : أَلْبَسْتُه الحَدِيدَ .<br>\
وقال أَبُو مَنْصُور : يُقَالُ لَهَا الضَّبَّةُ والكَتِيفَةُ ؛ لأَنَّهَا عَرِيضَةٌ كهَيْئة خَلْق الضَّبِّ ؛ وسُمِّيَت كَتيفَةً لأَنَّهَا عُرِّضتْ على هَيْئة الكَتِف .<br>\
وفي الأَسَاس : من المجاز : وعلى بَابِه ضَبَّةٌ وضَبَّاتٌ وضِبَابٌ .<br>\
وباب مُضَبَّبٌ ، ولِسِكِّينه ضَبَّةٌ : وهي الجُزْأَة لانَّها تَشُدُّ النَّصَاب ، انْتَهَى .<br>\
وهَذَا قَدْ أَغْفَلَه المُؤَلِّف .<br>\
( و ) ضَبَّة : ( ة بتِهامَة ) بِسَاحِل البَحْر مِمَّا يَلِي طَرِيقَ الشَّأْم .<br>\
( و ) ضَبَّة : ( نَاقَةُ الأَحْبَشِ بن قَلَعٍ ) الشَّاعِرِ ( العَنْبَرِيّ ) التَّمِيمِيّ .<br>\
( و ) ضَبَّةُ : حيٌّ من العَرَبِ .<br>\
و ( ضَبَّةُ بْنُ أُدَ : عَمُّ تَمِيم بنِ مُرّ ) بْنِ أُدِّ بْنِ طَابِخَةَ بْنِ الْيَاس بْنِ مُضَر .<br>\
وأَبْنَاءُ ضَبَّة ثَلَاثَةٌ : سَعْدٌ وسُعَيْد ، مُصَغَّراً ، وباسِلٌ .<br>\
الأَخِيرُ أَبُو الدَّيْلم ، والَّذي قبله لا عَقِبَ لَهُ فَانْحَصَر جِمَاعُ ضَبَّةَ في سَعْدِ بْنِ ضَبَّة ، وهُم جَمْرَةٌ من جَمَرَات الْعَرَب ومنهم الرِّبَابُ .<br>\
والضَّبّ أَيْضاً : القَبْضُ على الشَّيْءِ بالكَفِّ وعن ابْنِ شُمَيْل : التَّضْبِيبُ : شِدَّةُ القَبْضِ على الشيء كَيْلَا يَنْفَلِتَ مِنْ يَدِه .<br>\
يقال : ضَبَّبَ عليه تَضْبِيباً .<br>\
( وأَضَبَّ : صَاحَ ) وجَلَّبَ .<br>\
( و ) قيلَ : ( تَكَلَّم ) ، عن أَبِي زَيْد ، وقيل : إِذَا تَكَلَّم مُتَتَابِعاً .<br>\
أَو أَضَبَّ القَوْمُ : كَلَّم بعضُهُم بَعْضاً .<br>\
وعَنْ أَبِي حَاتِم : أَضَبَّ القَومُ إِذا تَكَلَّمُوا وأَفَاضُوا في الحَدِيثِ .<br>\
( و ) أَضَبَّ في الغَارَةِ : نَهَدَ و ( اسْتَغَارَ ) .<br>\
وأَضَبُّوا عَلَيْه إِذا أَكْثَرُوا عَلَيْه .<br>\
وفي الحديث : ( فَلَمَّا أَضَبُّوا عَلَيْه ) أَي أَكْثَروا .<br>\
( و ) أَضَبَّ الشيء : ( أَخْفَى ) إِيَّاهُ .<br>\
( و ) أَضَبَّ ( النَّعَمُ : أَقْبَلَ وَفِيه تَفَرُّقٌ ) .<br>\
والضَّبَبُ والتَّضْبِيبُ : تَغْطِيةُ الشَّيْء ودُخُولُ بَعْضِه في بَعْض .<br>\
( و ) أَضَبَّ ( الشَّعَرُ : كَثُرَ .<br>\
و ) أَضَبَّتِ ( الأَرْضُ : كَثُرَ نَبَتُها ) .<br>\
وعن ابن بُزُرْج : أَضَبَّت الأَرضُ بالنَّبَاتِ :<br>\
طَلَعَ نَبَاتُها جَمِيعَاً .<br>\
( و ) أَضَبَّ ( فُلَاناً ) أَو عَلَى الشَّيْءِ : ( لَزِمَه فَلَمْ يُفَارِقْه ) .<br>\
وأَصْل الضَّبِّ : اللُّصُوقُ في الأَرْض وقد تَقَدَّم .<br>\
( و ) أَضَبَّ ( عَلَيْه : أَمْسَكَه ) عَنْ أَبِي زَيْد .<br>\
وقَال أَبُو حَاتِم : أَضَبَّ القَوْمُ : سَكَتُوا وأَمْسَكُوا عَنِ الْحَدِيث .<br>\
( و ) أعضَبَّ ( عَلَى المَطْلُوبِ : أَشْرَفَ ) عَلَيْه ( أَن يَظْفَرَ بِه ) .<br>\
قال أَبُو مَنْصُور : وهَذَا من ضَبَأَ يَضْبَأُ ، ولَيْسَ مِنْ بَابِ المُضَاعَف .<br>\
وقَدْ جَاءَ بِهِ اللَّيْثُ في بَابِ المُضَاعَف ، قَالَ : والصَّوَابُ الأَوَّلُ وهو مَرْوِيٌّ عنِ الكِسَائِيّ ، كَذَا في لِسَانِ الْعَرَب .<br>\
( و ) أَضَبَّ ( السَّقاءُ : هُرِيقَ مَاؤُه من خُرْزَةٍ فيه ) أَوْ وَهْيَةٍ .<br>\
( و ) أَضَبَّ ( اليَوْمُ ) أَي ( صَارَ ذَا ضَبَابٍ ، بالفَتْح ، أَي نَدًى كالغَيْم ) وَقِيلَ كالغُبَار يَغْشَى الأَرْضَ بالغَدَوَاتِ ( أَو سَحَاب رَقِيق ) ، سُمِّي بِذَلِكَ لتَغْطِيَته الأُفُق ، وَاحِدته ضَبابة .<br>\
وقد أَضَبَّتِ السماءُ إِذا كان لها ضَبَاب وأَضَبَّ الغَيْمُ : أَطْبَقَ .<br>\
وقيل : الضَّبَابَةُ : سَحَابَة تُغَشِّي الأَرْضَ ( كالدُّخَانِ ) .<br>\
والجَمْع الضَّبَابُ .<br>\
وفي الحديث : ( كنتُ مع النبيّ صلى الله عليه وسلم في طريق مكَّة فأَتَتْنَا ضَبَابَةٌ فَرَّقَتْ بَيْنَ النَّاسِ ) .<br>\
هي البُخَارُ المُتَصَاعِد من الأَرْضِ في يَوْمِ الدَّجْن يَصِيرُ كالظُّلَّةِ يَحْجُبُ الأَبْصَار لظُلْمَتِهَا .<br>\
( و ) أَضَبَّ فُلَانٌ ( على مَا فِي نَفْسِهِ ) أعي ( سَكَت ) .<br>\
وقال الأَصْمَعِيُّ : أَضَبَّ فُلانٌ على ما في نَفْسِه أَي أَخْرَجَه .<br>\
وقال أَبُو حَاتم : أَضَبَّ القَومُ إِذا سَكَتُوا وأَمْسَكُوا عَنِ الحدِيث .<br>\
وأَضَبُّوا إِذا تَكَلَّمُوا وأَفَاضُا في الْحَدِيثِ ( ضِدٌّ ) أَي زَعَمُوا أَنَّه من الأَضْدَادِ .<br>\
( و ) أَضَبَّ ( القَوْمُ : نَهَضُوا في الأَمْرِ جَمِيعاً ) .<br>\
وفي التَّهْذِيب في آخِرِ العَيْنِ مَعَ الجِيمِ ، قَال مُدْرِكٌ الجَعْفَرِيُّ : يقَال : أَضَبُّوا لِفُلَانٍ أَي تَفَرَّقُوا في طَلَبِه .<br>\
وقد أَضَبَّ القوْمُ في بُغْيَتِهِم أَي في ضَالَّتِهِم أَي تَفَرَّقُوا في طَلَبَها .<br>\
( والضَّبِيبَةُ : سَمْنٌ ورُبٌّ يُجْعَلُ للصَّبِيّ في عُكَّةٍ ) يُطْعَمُه .<br>\
( و ) يقال : ( ضَبَّبَةُ : أَطْعَمَه إِيَّاه ) وضَبِّبُوا لصَبِيِّكم .<br>\
( والضَّبُوبُ ) كَصَبُورٍ : ( الدَّابَّةُ ) الَّتِي ( تَبُولُ و ) هي ( تَعْدُو ) .<br>\
وقال الأَعْشَى :<br>\
مَتى تَأْتِنا تَعْدُو بسَرْجِكَ لَقْوَةٌ<br>\
ضَبُوبٌ تُحَيِّينَا ورأْسُك مَائِلُ<br>\
وأَهْلُ الفِراسَة يَجْعَلُونَه من العُيُوب .<br>\
وقد ضَبَّت  تضِبُّ ضُبوباً .<br>\
( و ) في حديث مُوسَى وشُعَيْب عَلَيْهِما السَّلَامُ : ( لَيْسَ فِيهَا ضَبُوبٌ ولا تَعُولٌ ) .<br>\
الضَّبُوبُ : ( الشَّاةُ الضَّيِّقَةُ ) ثَقْبِ ( الإِحْلِيلِ ) .<br>\
وفي نُسْخَة ( النَّاقَة ) بَدَل ( الشَّاةِ ) ، والأُولى هِيَ الصَّوَابُ .<br>\
( و ) الضَّبُوبُ : ( فَرَسُ جُمَانَةَ ) بْنِ رَبِيعَةَ الحَارِثِيّ ) .<br>\
( و ) الضُّبَيْبُ ( كَزُبَيْرٍ : فَرَسَانِ لحَسَّانَ بْنِ حَنْظَلَةَ ) الطَّائِيِّ ( وحَضْرَمِيّ بْنِ عامِرٍ ) الأَسَدِيّ ، ولأَحَدِهما حَدِيثٌ .<br>\
( و ) ضُبَيْبٌ : ( مَاء .<br>\
ووَادٍ ) .<br>\

( والفَحَّاشُ الجَرِيءُ ) قال أَبو زيد : رَجُلٌ ضِبْضِبٌ ، وامرَأَةٌ ضِبْضِبَةٌ وَهُوَ الجَرِيءُ عَلَى مَا أَتَى ، وهُوَ الأَبْلَخُ أَيضاً ، وامرأَةٌ بَلْخَاءُ ، وهي الجَرِيئَةُ التي تَفْخَر عَلَى جِيرَانِهَا ( كالضُّبَاضِبِ ) كَعُلَابِط .<br>\
( وضَبِيبُ السَّيْفِ ) كأَمِير : ( حَدُّه ) ، ومثْلُه في تَّوْشِح ، وكَذَا ضَبَّةُ السَّيْف ، قالَه الخَطَّابِيّ ولمْ يَذْكُرْه ابنُ الأَثِير .<br>\
( ومَضَبٌّ ) بالفَتْح : ( ع ) .<br>\

( وسَمَّوْا ضَبَّا وضَبَّاباً وضِبَاباً ومُضِبًّا كشَدَّاد وكِتاب ومُحِبّ ) والضِّبابُ بالكَسْرِ : اسْم رجل ، وهُو أَبُو بَطْن سُمِّيَ بجَمْع الضَّبِّ .<br>\
