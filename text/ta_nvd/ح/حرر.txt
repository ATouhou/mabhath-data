مُسْتنُّ الحَرُورِ : مُشْتدُّ حَرِّها ؛ شَبَّه رَفْرَفَ الفُسْطَاطِ عند تَحرُّكِه لهُبُوب الرِّيحِ بسَبِيبِ الفَرسِ .<br>\
( و ) الحَرُورُ : ( حَرُّ الشَّمْسِ ) .<br>\
وقيل : الحَرُرُ : اسْتِيقادُ الحَرّ ولَفْحُه ، وهو يكونُ بالنَّهَار واللّيلِ ، والسَّمُوم لا يَكون إِلّا بالنَّهار .<br>\
( و ) في الكتاب العزيز : { وَلاَ الظّلُّ وَلاَ الْحَرُورُ } ( فاطر : 21 ) قال الزَّجّاج : معناه لا يَسْتوِي أَصحابُ الحَقِّ الذين هم في ظلَ من الحَقّ ، ولا أَصحابُ الباطِلِ الذين هم في الحَرُورِ ، أَي ( الحَرّ الدّائِم ) لَيلاً ونَهاراً .<br>\
( و ) قال ثعلب الظِّلُّ هنا الجَنَّةُ ، والحَرُرُ ( النّارُ ) .<br>\
قال ابن سِيدَه : والذي عندي أَن الظل هو الظِّلُّ بعَيْنِه ، والحَرُورَ الحَرُّ بعَيْنِه .<br>\
وجمعُ الحَرُورِ حَرائِر ، قال مُضَرِّس :<br>\
لَمَّاعَةٍ قد صادَفَ الصَّيْفُ ماءَها<br>\
وفاضَتْ عليها شَمْسُه وحَرائِرُهْ<br>\
( وحُريْر كزُبَيْرٍ ) أَبو الحُصَيْنِ ، ( شيخُ إِسحاقَ بنِ إِبراهيمَ المَوْصَلِيِّ ) النَّدِيمِ المشهورِ .<br>\
( وقَيْسُ بنُ عُبَيْدِ بنِ حُرَيْرِ ) بنِ عَبْدِ بنِ الجَعْدِ النَّجّارِيُّ المازِنِيُّ أَبو بَشِيرٍ : ( صَحَابِيٌّ ) ، قُتِلَ باليَمَامَةِ وَرَوَى عنه ضَمْرةُ بنُ سَعيدٍ .<br>\
وفَاتَ : عَمْرُو بنُ الحُرَيْرِ الأَسدِيّ ، أَخْبَارِيّ .<br>\
( والحُرِّيَّةُ ) ، بالضمّ : ( الأَرضُ الرَّمْلِيَّةُ اللَّيِّنَةُ ) الطَّيِّبَةُ الصالِحَةُ للنَّباتِ ، وهو مَجازٌ .<br>\
وفي الأَساس : أَرْضٌ حُرَّةٌ : لا سَبَخَةَ فيها .<br>\
( و ) مِن المَجاز : الحُرِّيَّةُ ( مِن العَربِ : أشرافُهُم ) ، يقال : ما في حُرِّيَّةِ العَرَبِ والعجَمِ مِثلُه ، وقال ذو الرُّمَّةِ :<br>\
فصارَ حَياً وطَبَّقَ بعد خَوْفٍ<br>\
على حُرِّيَّةِ العَرَبِ الهُزَالَى<br>\
أَي على أَشْرَافِهم .<br>\
ويقال : هو مِن حُرِّيَّةِ قَومِه : أَي مِن خالِصِهم .<br>\
والحُرُّ مِن كلِّ شيْءٍ : أَعْتَقُه .<br>\
( والحُرَيْرَةُ كهُرَيْرَةَ : ع قُرْبَ نَخْلَةَ ) بين الأَبْوَاءِ والجُحْفَةِ .<br>\
( وحُرَيْرٌ ، بالضمّ : د ، قُرْبَ آمِدَ ) ، كذا في النُّسَخِ ، والصَّوابُ : حُرِّينُ ، بالنُّون ، كذا في التَّكْمِلَة .<br>\
( وحَرُوراءُ ، كَلُولاءَ ) ، بالمَدِّ ، ( وقد تُقْصَرُ : ة بالكُوفَةِ على مِيلَيْن منها ، نَزَلَ بها جماعَةٌ خَالَفُوا عليًّا ، نَزَلَ بها جماعَةٌ خَالَفُوا عليًّا رضيَ اللّهُ عنه ، مِن الخَوَارِجِ .<br>\
( و ) يقال : ( هو -حَرُورِيٌّ بَيِّنُ الحَرُورِيَّةِ ) ، يَنْتَسِبُون إِلى هاذه القريَةِ ، ( وهم نَجْدَةُ ) الخارِجِيُّ ( وأَصحابُه ) ومن يعتقدُ اعتقادَهم ، يقال له : -الحَرُورِيُّ ، وقد وَرَدَ أَن عائسةَ رضيَ اللّهُ عنها قالتْ لبعضِ مَن كانَت تَقْطَعُ أَثَرَ دَمِ الحَيْضِ مِن الثَّوْب : أَحْرُورِيةٌ أَنتِ ؟ تَعْنِيهم ، كانوا يُبَالِغُون في العِبَاداتِ ، والمشهورُ بهاذه النِّسبة عِمْرَانُ بنُ حطّانَ السَّدُوسِيُّ الحَرُوريُّ .<br>\
ومن سَجَعَات الأَساس : ليس مِن الحُرُورِيَّةِ أَن يكونَ مِن الحَرُورِيَّةِ .<br>\
( و ) مِنَ المَجَازِ : ( تَحْرِيرُ الكِتَابِ وغيره تَقْوِيمُه ) وتَخْلِيصُه ؛ بإِقامةِ حُرُوفِه ، وتَحْسِينه بإِصلاحِ سَقطِه .<br>\
وتَحْرِيرُ الحِسَابِ : إِثباتُه مُسْتَوِياً لا غَلث فيه ، ولا سَقط ، ولا مَحْو .<br>\
( و ) التَّحْرِير ( للرَّقَبَةِ : إِعتاقُهَا ) .<br>\
والمُحَرَّرُ الذي جُعِلَ من العَبِيدِ حُرًّا فُعْتِقَ .<br>\
يقال : حَرَّ العَبْدُ يَحَرُّ حَرَارةً بالفتح أَي صار حُرًّا .<br>\
وفي حديث أَبِي الدَّرْدَاءِ : ( شِرارُكم الذين لا يُعْتَقُ مُحَرَّرُهُم ) ؛ أَي أَنهم إِذا أَعْتَقُوه اسْتَخْدَمُوه ، فإِذا أَرادَ فِراقَهم ادَّعَوْا رِقَّه .<br>\
( ومُحَرَّرُ بنُ عامرٍ ) الخَزْرَجِيُّ النَّجّارِيُّ ( كمُعَظَّمٍ : صَحابِيٌّ ) بدْرِيٌّ ، تُوُفِّيَ صَبِيحَةَ أُحُدٍ ، ولم يُعْقِبْ .<br>\
( و ) مُحَرَّرُ ( بنُ قَتَادَةَ كان يُوصِي<br>\
بَنِيه بالإِسلام ) ، ويَنْهَى بَنِي حَنِيفَةَ عن الرِّدَّةِ ، وله في ذالك شِعْرٌ حَسَنٌ أَورَدَه الذَّهَبِيُّ في الصَّحَابَة .<br>\
( و ) مُحَرَّرُ ( بنُ أَبي هُرَيْرَةَ : تابِعِيٌّ ) ، يَرْوِي عن أَبِيه ، وعنه الشَّعْبِيّ ، وأَهلُ الكُوفَةِ .<br>\
ذَكَرَه ابنُ حِبّانَ في الثِّقات .<br>\
( ومُحَرَّرُ دارِمٍ : ضَرْبٌ مِن الحَيّاتِ ) ، نَقَلَه الصَّغانِيُّ .<br>\
( و ) مِنَ المَجاز : ( اسْتَحَرَّ القَتْلُ ) في بني فلانٍ : إِذا ( اشتَدَّ ) وكَثُرَ ، كحَرَّ ، ومنه حديثُ عليَ رضيَ اللّهُ عنه : ( حَمِسَ الوَغَى واسْتَحَرَّ الموتُ .<br>\
( و ) يقال : ( هو أَحَرُّ حُسْناً منه ) ، وقد جاءَ ذالك في الحديث : ( ما رأَيتُ أَشْبَه برسولِ اللّهِ صلَّى اللّهُ عليْه وسلّم من الحَسَن ؛ إِلّا أَن النبيَّ صلَّى اللّهُ عليْه وسلَّم كان أَحَرَّ حُسْناً منه ) ؛ ( أَي أَرَقَّ منه رِقَّةَ حُسْنٍ ) .<br>\
( والحارُّ مِن العَمَلِ : شاقُّه وشديدُهُ ) وقد جاءَ في الحديث عن عليَ ( أَنه قال لفاطمةَ رضيَ للّهُ عنهما : لو أَتَيْتِ النبيَّ صلَّى اللّهُ عليْه وسلَّم فسَأَلْتِ خادِماً تَقِيكِ حارَّ ما أَنتِ فيه مِن العَملِ ) .<br>\
وفي أُخرى : ( حَرَّ ما أَنتِ فيه ) ؛ يَعْنِي التَّعَبَ والمَشَقَّةَ ، مِن خِدْمَةِ البيتِ ؛ لأَن الحَرَارَةَ مَقْرُونةٌ بهما ، كما أَن البَرْدَ مقرونٌ بالرّاحَةِ والسُّكُونِ .<br>\
والحارُّ : الشاقُّ المُتْعِبُ ، ومنه الحديثُ الآخَرُ عن الحَسَنِ بنِ عليَ : ( قال لأَبِيه لمّا أَمَره بجلْدِ الولِيدِ بنِ عُقْبَةَ ؛ وَلِّ حارَّها مَن تَوَلَّى قارَها ) ؛ أَي وَلِّ الجَلْدَ مَن يَلْزَمُ الوَلِيدَ أَمرُه ، ويَعْنِيه شَأْنُه .<br>\
( و ) الحارُّ : ( شَعرُ المَنْخرَيْنِ ) ؛ لما فيه مِن الشِّدَّةِ والحَرَارَة ، نقلَه الصغانيُّ .<br>\
( وأَحَرَّ النّهَارُ : صارَ حَارًّا ) ، لغةٌ في حَرَّ يَومُنَا ، سَمِع ، الكِسَائيُّ ، وحَكاهما ابنُ القَطّاعِ في الأَفْعَال والأَبْنِيَةِ ، والزَّجّاجُ في : فَعَلت وأَفْعَلت ، قال<br>\
شيخُنا : ومثلُ هاذا عند حُذّاقِ المُصَنِّفِين مِن سُوءِ الجَمْعِ ؛ فإِنّ الأَوْلَى التَّعَرّضُ لهاذا عند قوله : ( حَرَرْتَ يا يَومُ ) ، بالوجُوهِ الثلاثةِ ، وهو ظاهرٌ .<br>\
( و ) أَحَرَّ ( الرجلُ : صارتْ إِبلُه حِراراً ، أَي عِطَاشاً ) .<br>\
ورجلٌ مُحِرٌّ : عَطِشَتْ إِبلُهُ .<br>\
( وحَرْحَارٌ ) ، بالفتح : ( ع ببلادِ جُهَيْنَةَ ) بالحِجاز .<br>\
( ومحمّدُ بنُ خالدٍ ) الرّازِيُّ ( -الحَرَوَّرِيُّ كعَمَلَّسِيَ محدِّث ) ، وقال السَّمْعَانِيُّ : هو أَحمدُ بنُ خالدٍ ، حدَّث عن محمّدِ بنِ حُمَيْدٍ ، وموسَى بنِ نصرٍ الرّازِيَّيْنِ ، ومحمّدِ بنِ يَحْيَى ، ومحمّدِ بنِ يَزيِدَ السُّلَمِيِّ النَّيْسَابُورِيَّيْنِ ، رَوَى عنه الحُسَيْنُ بنُ عليَ المعروفُ بحُسَيْنك ، وعليُّ بنُ القاسم بن شاذانَ ، قال ابن ماكُولَا : لا أَدْرِي : أَحمد بن خالد الرازيّ الحَرورِيُّ إِلى أَيِّ شيْءٍ نُسِبَ .<br>\
قلْت : وهاكذا ذَكَرَه الحافظُ في التَّبْصِير أَيضاً بالفَتْح ولم يذكر أحدٌ منهم أَنَّه الحَرَوَّرِيُّ ، كعَمَلَّسِيَ ، ففي كلام المصنِّف مَحَلُّ تَأَمُّلٍ .<br>\
وممّا يُستدرَك عليه :<br>\
الحَرَرُ محرَّكةً أَنْ يَيْبَسَ كَبِدُ الإِنسانِ ، مِن عَطَشٍ وحُزْنٍ .<br>\
والحَرُّ : حُرْقَةُ القَلبِ ، مِن الوَجَع والغَيْظِ والمَشَقَّةِ .<br>\
وأَحَرَّها اللّهُ .<br>\
والعربُ تقول في دُعائها على الإِنسان : مالَه أَحَرَّ اللّهُ صَدْرَه ؛ أَي أَعْطَشَه ، وقيل : معناه أَعْطَشَ اللّهُ هامتَه .<br>\
ويقال : إِنِّي أَجِدُ لهاذا الطَّعَمِ حَرْوَةَ فِي فَمِي ، أَي حَرارةً ولَذْعاً ، والحَرَارَةُ : حُرْقَةٌ في الفَمِ مِن طَعْم الشَّيْءِ ، وفي القَلْب مِن التَّوَجُّع ، مِن ذالك قولُهم : وَجَدَ حَرارَةَ السَّيْفِ ، والضَّرْبِ ، والمَوْتِ ، والفِرَاقِ ، وغيرِ ذالك ، نَقَلَه ابن دُرُسْتَوَيْهِ ، وهو من الكِنَايَاتِ ،<br>\
والأَعْرَفُ الحَرْوَةُ ، وسيأْتي في المعتلّ .<br>\
وقال ابن شُمَيْلٍ : الفُلْفُلُ له حَرارةٌ وحَرَاوَةٌ ، بالراءِ والواو .<br>\
والحَرَّةُ : حَرَارةُ في الحَلْق ، فإِن زادتْ فهي الحَرْوَةُ ، ثم الثَّحْثَحَةُ ، ثم الجَأْرُ ، ثم الشَّرَقُ ، ثم الفُؤُقُ ، ثم الحَرَضُ ، ثمَّ العَسْفُ ، وهو عند خُرُوجِ الرُّوحِ .<br>\
واسْتَحْرَرْتُ فُلانةَ فحَرَّتْ لُي ؛ أَي طَلَبْتُ منها حَرِيرَةً فعَمِلَتْها .<br>\
وفي حديث أَبي بَكْ : ( أَفمنكم عَوْفٌ الذي يُقال فيه : لا حُرَّ بوادِي عَوْف ؟ قال : لا ) .<br>\
هو عَوْفُ بنُ مُحَلِّمِ بنِ ذُهْلٍ الشَّيْبَانِيُّ ، كان يقال له ذالك لِشَرَفِه وعِزِّه ، وأَنْ مَن حَلَّ بوادِيه مِن النّاس كان له كالعَبِيدِ والخَوَلِ والمُحَرَّرُ ، كمُعَظَّمٍ : المَوْلَى ، ومنه حديثُ ابنِ عُمَرَ ، أَنه قال لمُعَاوِيَةَ رضيَ اللّهُ عنهم : ( حاجَي عَطَاءُ المُحَرَّرِينَ ) ؛ أَي المَوالِي ، أَي المَوالِي ، أَي لأَنهم قومٌ لا دِيوانَ لهم ؛ تَأَلُّفاً لهم على الإِسلام .<br>\
وتَحْرِيرُ الوَلَدِ أَن يُفْرِدَه لطاعةِ اللّهِ عَزَّ وجَلَّ ، وخِدْمةِ المَسْجِدِ .<br>\
وقولُه تعالَى حكايةً عن السَّيِّدة مَرْيَمَ ابْنَةِ عِمْرَانَ : { إِنّي نَذَرْتُ لَكَ مَا فِي بَطْنِي مُحَرَّرًا } ( آل عمران : 35 ) قال الزَّجّاج : أَي خادِماً يَخْدُم في مُتَعَبَّداتِكَ ، والمُحَرَّرُ : النَّذِيرُ .<br>\
والمُحَرَّرُ : النَّذِيرَةُ .<br>\
وحَرَّرَه : جَعَلَه نَذِيرَةً في خِدْمةِ الكَنِيسَةِ ما عاشَ لا يَسَعُهُ تَرْكُها في دِينه .<br>\
ومن المَجاز : أَحرارُ البُقُولِ : ما أُكلَ غَيرَ مطْبُوخٍ ، واحِدُهَا حُرٌّ ، وقيل : هو ما خَشُنَ منها ، وهي ثلاثةٌ : النَّفَلُ ، والحُرْثُبُ ، والقَفْعاءُ .<br>\
وقال أَبو الهَيْثَم : أَحرارُ البُقُولِ : ما رَقَّ منها ورَطُبَ ، وُكُورُها : ما غَلُظَ منها وخَشُنَ .<br>\
وقيل : الحُرُّ : نَباتٌ مِن نَجِيلِ السِّباخِ .<br>\
والحُرَّةُ : البابُونَجُ .<br>\
والحُرَّة : الوَجْنَةُ .<br>\
والحُرَّتانِ : الأُذُنانِ ، ومنه قولُهم : حَفِظَ اللهُ كَرِيمَتَيْكَ وحُرَّتَيْكَ ، وهو مَجازٌ .<br>\
وحَرَّ الأَرْضَ يَحَرُّهَا حَرّاً : سَوّاهَا .<br>\
والمِحَرُّ : شَبَحَةٌ فيها أَسْنانٌ ، وفي طَرْفِها نَقْرَانِ ، فيهما عُودٌ مَعْطُوفٌ ، وفي وَسَطِها عُودٌ يُقْبَضُ عليه .<br>\
