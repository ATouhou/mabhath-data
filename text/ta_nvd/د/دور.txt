هاكذا هو مضبوط ولم يَذكره المُصَنّفِ في مَحَلّه ، والصَّواب أَنه مُصَغّر ، جُدَيّ ، وهو جَبَلٌ نَجديٌّ في ديار طيِّىء .<br>\
( و ) دارَةُ ( جُلْجُلٍ ) ، كقُنْفُذ ، بنَجْد ، في دارِ الضِّبابِ ، مما يُواجِهُ دِيَارَ فَزَارَةَ ؛ قد جاءَ ذكره في لاميّة امْرِىء القَيْس .<br>\
( و ) دارَةُ ( الجَلْعَبِ ) : مَوضعٌ في بِلادِهِم .<br>\
( و ) دارَةُ ( الجُمُد ) ، كعُنُق : جَبَلٌ بنَجْد ، مثَّلَ به سيبويهِ ، وفسّره السّيرافيّ ، وقد تقدم ، وضَبَطه الصَّغَانيّ بفَتْح فسكون .<br>\
( و ) دارَةُ ( جَوْداتٍ ) ، بالفتح ، ولم يَذكره المُصَنِّف في مَحَلّه ، والأَشبهُ أَن يكون ببلاد طَيِّىءٍ .<br>\
( و ) دارَةُ ( الجَوْلاءِ ) ، ولم يَذكره المصنّف في اللَّام .<br>\
( و ) دارَةُ ( جَوْلَةَ ) ، ولم يَذكره المصنّف في اللام .<br>\
( و ) دارَةُ ( جُهْد ) ، بضَمّ فسُكُون .<br>\
( و ) دارَةُ ( جَيْفُون ) ، بفتح الجيم وسكون التحتيّة وضَمّ الفاءِ .<br>\
( و ) في حرف الحاءِ اثْنَتَانِ :<br>\
دارَةُ ( حُلْحُلٍ ) ، كقُنْفُذٍ ، ( وليس بِتَصْحِيف جُلْجُل ) ، كما زَعمه بعضُهم ، ونمهم من ضَبطه كجَعْفَر .<br>\
وقال هو جَبَلٌ من جِبَالِ عُمَانَ .<br>\
( و ) دارَةُ ( حوْقٍ ) ، بفتح فسكون .<br>\
( و ) في حرف الخاءِ سبعة :<br>\
دارَةُ ( الخَرْج ) ، بفَتْح فسكُون ، باليَمامَة .<br>\
فإِن كان بالضَّمّ فهو في دِيَار تَيْم لِبني كَعْب بن العَنْبَر بأَسافِلِ الصَّمَّان .<br>\
( و ) دارة ( الخَلاءَةِ ) ، كسَحَابَةٍ ، وهو مُسْتَدْرك على المُصَنِّفَ في حَرْف الهمْزة .<br>\
( و ) دارَةُ ( الخَنَازِير ) .<br>\
( و ) دارَةُ ( خَنْزَرٍ ) ، كجَعْفر ، ويُكْسَر ، هاذِه عن كُراع .<br>\
قال الجَعْدِيّ :<br>\
أَلمَّ خَيَالٌ من أُمَيْمَةَ مَوْهِناً<br>\
طُرُوقاً وأَصْحَابي بدَارَةِ خَنْزَرِ<br>\
( و ) دارَةُ ( الخَنْزَرَتَيْنِ ) تَثنِية<br>\
خَنْزَرة ، وفي بعض النُّسخ الخَزْرتين .<br>\
( و ) دارَةُ ( الخِنْزِيرَين ) تَثْنِيَة خِنْزِير .<br>\
وفي التكملة : دارة الخِنْزِيرَتَيْنِ .<br>\
ويقال : إِن الثانِيَةَ روايَةٌ في الأَولَى ، وقد تقدّم ذالك في ( خ ز ر ) وفي ( خ ن ز ر ) .<br>\
( و ) دارَةُ ( خَوَ ) : واد يَفْرُغ ماؤُه في ذِي العُشَيْرَةَ من ديار أَسدٍ لبني أَبي بَكْرِ بنِ كِلاب .<br>\
( و ) في حرف الدال أَربعةٌ :<br>\
دارَةُ ( داثِرٍ ) : ماء لفَزَارةَ ، وهو مستدرك على المُصَنِّف في ( د ث ر ) .<br>\
( و ) دارة ( دَمْخٍ ) ، بفَتْح فكسون ، وهو جَبَل في دِيَارِ كِلاب ، وقد تقدّم .<br>\
( و ) دارَةُ ( دَمُّون ) ، كتَنُّور : مَوضع سيأْتي ذِكْرُه .<br>\
( و ) دارَة ( الدُّور ) ، بالضَّمّ : موضع بالبادية ، قال الأَزهريّ : وأُراهم إنما بالَغُوا بها كما تقول رَمْلة الرِّمان .<br>\
( و ) في حرف الذال ثلاثةٌ :<br>\
دارَةُ ( الذِّئْب ) ، بنجد في دِيَار كِلاب .<br>\
( و ) دارَةُ ( الذُّؤَيْب ) ، بالتَّصْغِير ، لبني الأَضْبَط ، وهما دَارَتان ، وقد تقدّم ذِكْرهُما .<br>\
( و ) دارَةُ ( ذات عُرْش ) ، بضمّ العين المهملة وسكون الرُّاءِ وآخرُه شِينٌ مُعْجَمة ، وضَبَطه البَكْرِيّ بضَمَّتَيْن : مَدِينَة يَمانيَة ، على الساحل ، ولم يَذْكُره المصنِّف ، وما إِخال البكريَّ عَنَى هاذِه الدَّارةَ .<br>\
( و ) في حرف الراءِ تسعة .<br>\
دارَةُ ( رابِغ ) : وادٍ دُون الجُحْفَة على طريق الحَاجِّ من دون عَزْوَرٍ .<br>\
( و ) دارَةُ ( الرَّجْلَيْن ) ، تَثنيَة رَجْل بالفتْح ، لبني بَكْرِ بنِ وائِل من أَسافلِ الحَزْن وأَعالي فَلْج .<br>\
( و ) دارَةُ ( الرَّدْمِ ) ، بفَتْح فسكُون ، وضَبَطه بعضُهم بالكَسْر : موضعٌ يأْتِي ذِكْرُه في المِيمِ .<br>\
( و ) دارَةُ ( ردهة ) ؛ وهي حُفَيْرة<br>\
في القُفّ وهو اسم مَوْضِع بعَيْنه ، وسيأْتي في الهاءِ ولم يذكُره المصنّف .<br>\
( و ) دارَةُ ( رَفْرَفٍ ، بمهْمَلَتَين مَفْتُوحَتَيْنِ ) وتُضَمَّان ، ونقله ياقوت عن ابنِ الأَعْرَابِيّ ، لبَنِي نُمَيْر ، ( أَو بمُعجَمتين مَضْمُومَتيْن ) ، والأَول أكثر .<br>\
( و ) دارةُ ( الرُّمْح ) ، بضَمّ الرّاءِ وسُكُون المِيمِ ، وضَبَطه بعضُهم بكسْرِ الرّاءِ ، أَبْرق في دِيَار بني كِلاب ، لبني عَمْرو بن رَبِيعة ، وعنده البَتِيلة ، ماءُ لهم ، وفي بَعْض النُّسَخ : الرِّيح ، بدل الرُّمح ، وهو غَلَط .<br>\
( و ) دارَةُ ( الرِّمْرِمِ ) ، كسِمْسِم : موضِع يأْتِي ذكرُه في المِيمِ .<br>\
( و ) دَارَةُ ( رَهْبَى ) ، بفَتْح فسُكُون وأَلف مقصورة : مَوضع ، وقد تَقدَّم ذِكْرَه .<br>\
( و ) دَارَةُ ( الرُّهَى ) ، بالضَّمِّ ، كهُدى وسيأْتِي ذِكْره .<br>\
( و ) في حرف السين اثنتان :<br>\
دارَةُ ( سَعْر ) ، بالفَتْح ( ويُكْسَر ) ، جاءَ ذِكْرُه في شِعر خُفاف بنِ نَدْبَةَ .<br>\
( و ) دارَةُ ( السَّلَم ) ، محرّكة .<br>\
( و ) في حرف الشِّين اثْنَتان :<br>\
دارَةُ ( شُبَيْثٍ ) ، مُصَغَّراً : موضع بنَجْد لبني رَبِيعة .<br>\
( و ) دارةُ ( شَجَا ، بالجِيم ، كقَفا ) : ماء بنجد في دِيَار بَنِي كِلاَب ( وليس بتَصْحِيفِ وَشْحَى ) كسَكْرَى .<br>\
( و ) في حرف الصاد أَربعةٌ :<br>\
دارَةُ ( صارَةَ ) : جَبَل في ديار بني أَسَد .<br>\
( و ) دارَةُ ( الصَّفَائِحِ ) : مَوضع تقدّم ذِكْره في الحاء .<br>\
( و ) دارَةُ ( صُلْصُل ) ، كقُنْفَذ : ماء لبني عَجْلانَ قُرْبَ اليمامَة ، وماء آخَر في هَضبةٍ حَمْراءَ لبني<br>\
عَمْرِو بن كِلاب في دِيَاره بنَجْد .<br>\
( و ) دارَةُ ( صَنْدَلٍ ) : مَوضِع ، وله يَوم معروف ، وسيَأْتي ذِكْرُه .<br>\
( و ) في حَرف العَيْن سبعةٌ :<br>\
دارَةُ ( عَبْسٍ ) ، بفتح فَسُكُون : ماء بنَجْد في دِيَار بني أَسَد .<br>\
( و ) دارَةُ ( عَسْعَسٍ ) : جَبَل لبني دُبَيْر في بلاد بَنِي جَعْفَر بنِ كِلابٍ ، وبأَصْله ماءُ النَّاصِفةِ .<br>\
( و ) دارَةُ ( العَلْيَاءِ ) ، وهو مُسْتَدْرك على المُصَنّف في المعتلّ .<br>\
( و ) دارَةُ ( عُوَارِضٍ ) ، بالضّمّ : جَبل أَسودُ في أَعلَى دِيارِ طَيِّىءٍ ، وناحية دارِ فَزَارةَ .<br>\
( و ) دارَةُ ( عُوَارِمٍ ) ، بالضّمّ : جَبل لأَبي بكر بن كلاب .<br>\
( و ) دارَةُ ( العُوجِ ) ، بالضّمّ : مَوْضِع باليَمَن .<br>\
( و ) دارَةُ ( عُوَيْجٍ ) ، مُصَغّراً :<br>\
موضع آخَر .<br>\
مَرَّ ذِكْرُهما في الجِيمِ .<br>\
( و ) في حرف الغين ثلاثةٌ :<br>\
دارَةُ ( الغُبَيْرِ ) ، مُصَغَّراً : ماء لبني كِلاب ، ثمّ لِبَنِي الأَضبطِ بنَجْد ، وماء لمُحَارِبِ بن خَصَفَةَ .<br>\
( و ) دارَةُ ( الغُزَيِّلِ ) مُصَغَّراً ، لِبَلْحَارِثِ بنِ ربيعةَ ، كما سيأْتي .<br>\
( و ) دارَةُ ( الغُمَيْرِ ) ، مُصَغَّراً : في ديارِ بَنِي كِلاب عند الثَّلَبُوتِ .<br>\
( و ) في حرف الفاءِ ثلاثةٌ :<br>\
دارَةُ ( فَتْكٍ ) ، بفَتْح فسُكُون ، وضَبطَه البَكْرِيّ بالكَسْر : مَوضع بينَ أَجَأَ وسَلْمَى .<br>\
( و ) دارَةُ ( الفُرُوعِ ) ، جَمْع فَرْع : مَوضع مُسْتَدْرَك على المُصَنِّف .<br>\
( و ) دارَةُ ( فَرْوَعٍ ، كجَرْوَل ) : مَوضِع آخَر ، ( وهي غير دَارَةِ الفُرُوعِ ) .<br>\
( و ) في حرف القاف تِسْعَةٌ :<br>\
دارَةُ ( القِدَاحِ ، ككِتَاب ) .<br>\
( و ) دارَةُ القَدَّاحِ ، مثْل ( كَتّانٍ ) ، من ديار بني تَمِيم .<br>\
وهما دارَتَان .<br>\
( و ) دارَةُ ( قُرْحٍ ) ، بضمّ فَسُكون بوادِي القُرَى .<br>\
وفي بعض النُّسخ ، قُرْط ، بدل ، قُرْح .<br>\
( و ) دارَةُ ( القُِطْقُِطِ ، بكسرتين وبضَمَّتيْنِ ) ، هاكذا ضَبَطَه بالوجْهين في حرف الطاءِ ، وسيأْتي هناك .<br>\
( و ) دارَةُ ( القَلْتينِ ) ، بفتح القاف وسكون اللام وكسر المثنّاة الفَوقِيّة ، وضبطه ياقوت بفتح المُثَنَّاة على الصّواب ، وهو ناحيَة باليَمَامة ويُقال لها : ذاتُ القَلْتَيْن ، ومنهم من ضَبَطَه بضمّ القاف وهو غَلَطٌ ، وقد سبق الكلام عَليه .<br>\
( و ) دارَةُ ( القِنَّعْبَةِ ) ، بكسر القاف وتَشْدِيد المَفْتُوحَة وسُكون العَيْن المهملة وفتح البَاءِ الموحدة ، وهو مُسْتدْرَك على المُصَنِّف في حَرْف الباءِ .<br>\
( و ) دارَةُ ( القَمُوصِ ) ، كصَبور : بقُرْب المدينة المُشْرَّفَة ، على ساكنِها أَفضلُ السلامِ .<br>\
( و ) دارَةُ ( قَوَ ) : بين فَيْدٍ والنِّبَاجِ .<br>\
( و ) في حرف الكاف خَمْسَةٌ :<br>\
دارَةُ ( كامِسٍ ) ، مَوضع سيأْتي ذِكرُه في السِّين .<br>\
( و ) دارَةُ ( كِبْدٍ ) ، بكَسْر فسُكُون ، وضبطَه البَكْرِيّ بكَسْر المُوَحَّدةِ أَيضاً ؛ وهي هَضْبةٌ حمراءُ بالمَضْجَع من ديار كِلاب .<br>\
( و ) دارَةُ ( الكَبْسَاتِ ) ، بفَتْح فسُكُون ، هاكذا هو مَضْبُوط ، والذي ذكرَه يَاقُوت والبَكْرِيّ : الكَبِيسَتان شبِيكتان لبني عبسٍ لهما وادياً<br>\
النفاخين حيث انقطعت حلّة النباج والْتقتْ هي ورَملة الشَّقيق .<br>\
والمصنّف لم يَذكر في السين لا الكبسات ولا الكبيستان فلينظر .<br>\
( و ) دارَةُ ( الكَوْرِ ) ، بفَتْح فسُكونٍ : جبل بين اليمامَةِ ومكّة لبَنِي عامرٍ ثم لبَنِي سَلُولَ .<br>\
( و ) دارَةُ ( الكُورِ ) ، بالضَّمِّ ، ( وهي غَيْرُ الأُولَى ) ، في أَرض اليَمَن ، بها وَقْعَة ، ويقال لها أَيضاً ثَنِيَّة الكُورِ .<br>\
( و ) في اللام واحدةٌ ، وهي :<br>\
دارَةُ ( لاقِطٍ ) ، لم يَذْكُره في الطَّاءِ ، وسيأْتي الكلام عليه .<br>\
( و ) في حرف المِيمِ سِتَّةَ عشَرَ : وهي دارة ( مَأْسَل ) ، كمَقْعَد مهموزاً ، سيأْتي للمصّنف في أَسل .<br>\
( و ) دارة ( مُتَالِع ) ، بالضَّمّ : جَبَل في بلاد طَيِّىء مُلاصِق لأَجأَ ، وقيل لبني صَخْرِ بن جَرْم .<br>\
وفي أَرض كِلاب يبن الرُّمَّةِ وضَرِيّة ، وأَيضاً شِعْب فيه نَخلٌ لبين مُرَّة بْنِ عَوْف .<br>\
وقيل : في دِيار بني أَسَد ، وسيأْتي في حَرْفِ العَيْن .<br>\
( و ) دَارَةُ ( المَثَامِنِ ) لبَنِي ظَالِمِ بن نُمَيْر .<br>\
( و ) دارَةُ ( المَرَاضِ ) ، كسَحَاب : مَوضع لهُذَيْل .<br>\
( و ) دارَةُ ( المَرْدَمَةِ ) ، بالفتح : لِبَنِي مَالِكِ بنِ رَبِيعَة .<br>\
( و ) دارَةُ ( المَرْوَرَاتِ ) ، بفتح فَسُكُون ، كأَنَّه جمع مَرْوَرٍ ، كجَعْفَر ، وسيأْتي ذِكْره .<br>\
( و ) دارَةُ ( مَعْرُوفٍ ) : ماء لبني جَعْفَر .<br>\
( و ) دارَةُ ( مُعَيطٍ ) ، كزُبَيْر ، وقيل كأَمِير : مَوضع يأْتي ذِكْره .<br>\
( و ) دَارَةُ ( المَكَامِنِ ) ، وسيأْتي للمُصَنّف في النون أَنه دارةُ المَكَامِين ، وأَنَّه لُغَة في الذي بعده .<br>\
( و ) دَارَةُ ( مَكْمَن ) ، كمَقْعَد ، ويقال : المَكَامِين ، في بِلاد قَيْس .<br>\
قال الرَّاعِي :<br>\
بدارَةِ مَكْمَنٍ ساقَتْ إِليها<br>\
رِياحُ الصَّيْفِ آرَاماً وعِينَا<br>\
( و ) دارَةُ ( مَلْحُوبٍ ) : ماء لبَنِي أَسَدِ بن خُزَيْمَة ، وقد تقدّم .<br>\
( و ) دارَةُ ( المَلِكَة ) ، أُنثَى المَلِك ، ولم يَذْكرُها ياقُوت في المُعْجَم ، وسيأْتي ذِكْرُهَا .<br>\
( و ) دَارَةُ ( مَنْوَرٍ ) ، كمَقْعَد : جَبل .<br>\
قال يَزِيد بنُ أَبِي حَارِثَةَ :<br>\
إِنّي لعَمْرُك لا أُصَالِحُ طَيِّئاً<br>\
حتّى يَغورَ مَكَانَ دَمْخٍ مَنْوَرُ<br>\

دارَةُ ( النّشَّاشِ ) ، ككَتّان ، هكَذا هو في سائِر النُّسَخ ، وضَبَطه ياقُوت في المُعْجَم النَّشْنَاش ، بزيادة نون ثَانِيَة بَعْد الشين .<br>\

( و ) في حرف الواو أَربعةٌ :<br>\

( و ) دارة ( وَسطٍ ) ، بفتح فسكون ( ويُحَرَّك ) : جَبل ضَخْم على أَربعة أَميال وَراءَ ضَرِيَّة لبني جَعْفَر بنِ كِلاب .<br>\
( و ) دارَةُ ( وشْحَى ) ، بالفَتْح ، ( ويُضَم ) ، وضَبَطه ياقوت بالمَدّ : ماء بنَجْد في دِيار بَنِي كِلاب .<br>\
( و ) في حرف الهاءِ واحدة :<br>\
دارَةُ ( هَضْبٍ ) ، بفتح فسكون ، قُربَ ضَرِيَّةَ من دِيَار كِلاب ، وقد تقدَّم ، وقيل لِلضِّباب .<br>\
( و ) في حرف الياءِ اثنتان<br>\
دارَةُ ( اليَعْضِيد ) ، وهو مُسْتَدْرك على المُصَنّف في الدّال ، ولم يَذْكُره ياقوت أَيضاً .<br>\
( و ) دارَةُ ( يَمْغُون ) بالغِين ( أَو يَمْعُون ) ، بالعَيْن المهملة ، وهو الذي صَرَّحَ به ياقُوت والبَكْرِيّ : من مَنازل هَمْدانَ باليمن .<br>\
وفي التَّكْمِلَة : دارَةُ يَمْعون أَو يَمْعُوز ، الأُولى بالنُّون والثانية بالزَّاي ، والعَيْن مُهْملة فيهما ، فتأَمَّل .<br>\
وهاذه آخِر الدارات ، وقد استوفَيْنَا بَيانَهَا على حَسَب ضِيقِ الوَقْتِ وقِلَّة المُساعد ، والله المُسْتَعان وعليه التُّكْلان .<br>\
( ودارَ ) الشْيءُ يَدُورُ ( دَوْراً ) ، بفَتْح فسُكُون ، ( ودَوَرَاناً ) ، مُحَرَّكةً ، ودُوُوراً ، كقُعلأد ، ( واسْتَدارَ ، وأَدَرْتُهُ ) أَنا ( ودَوَّرْتُه ، و ) أَدارَه غيرُه ودَوَّر ( به ) ، ودُرْت به ، ( وأَدَرْتُ : استَدَرْتُ ) .<br>\
وفي الحديث : ( إِن الزَّمَان قد استدَارَ كَهَيْئته يومَ خَلَقَ اللهاُ السَّماواتِ والأَرضَ ) ، يقال : دَارَ يَدُورُ<br>\
واستدَارَ  يَسْتَدِير ، إِذا طاف حَوْلَ الشَّيْءِ ، وإِذا عادَ إِلى المَوضع الذي ابتدأَ منه .<br>\
ومعنَى الحَدِيث أَنَّ العَرب كانوا يُؤَخِّرُون المُحَرَّم إِلى صفَر ، وهو النَّسِيءْ ، ليُقاتلوا فيه ، ويَفْعَلُون ذالك سَنَةً بعد سَنَةٍ ، فينْتَقِل المُحَرَّم من شَهْر إلى شَهْر ، حتى يَجْعَلُوه في جَمِيع شُهُور السَّنَة ، فلما كان تلك السنة كان قد عادَ إِلى زَمَنِه المَخْصُوص به قبل النَّقْل ودارَت السَّنَةُ كَهَيْئتِهَا الأُولَى .<br>\
( ودَاوَرَه مُدَاوَرَةً ودِوَاراً ) ، الأَخير بالكَسْر : ( دَارَ معه ) ، قال أَبو نُؤَيْب :<br>\
حتَّى أُتِيحَ له يَوماً بمَرْقَبَةٍ<br>\
ذُو مِرَّةٍ بِدِوَارِ الصَّيْدِ وَجَّاسُ<br>\
( والدَّهْرُ دَوَّارٌ به -وَدَوَّارِيٌّ ) ، أَي ( دَائِرٌ ) به ، على إضافة الشْيءِ إلى نفْسه .<br>\
قال ابنُ سِيده : هاذا قول اللُّغَوِيّين ، قال الفارِسِيّ : هو على لَفْظَ النَّسَب وليس بِنَسَبٍ ، ونَظِيرهُ بُخْتِيّ وكُرْسِيّ ، ومن المُضَاعَف أَعْجَمِيٌّ في مَعْنَى : أَعْجَم .<br>\
وقال اللَّيث : -الدَّوَّارِيُّ : الدَّهرُ بالإِنْسَان أَحوالاً .<br>\
قال العَجَّاجُ :<br>\
والدَّهْرُ بالإِنْسَان دَوَّارِيُّ<br>\

( والدُّوَارُ ، بالضَّم وبالفَتْح : شِبْه الدَّوَرَانِ يأْخُذُ في الرُّأْسِ .<br>\
و ) يقال : ( دِيرَ به ، و ) دِيرَ ( عليه ، وأُدِيرَ به : أَخَذَه ) .<br>\
وفي الأَساس : أَصابه الدُّوارُ .<br>\
من دُوَارِ الرأْس .<br>\
( ودُوَّارَةُ الرَّأْسِ ، كرُمّانة ويُفْتَح : طائِفَةٌ منه مستديرةٌ .<br>\
و ) الدُّوَّارَة ( من البَطنِ ) ، بالضمّ والفَتْح عن ثَعْلَب : ( ما تَحَوَّى من أَمعاءِ الشّاةِ ) .<br>\
( والدَّوَّارُ ، ككَتَّانِ ، ويُضَمُّ :<br>\
الكَعْبَةُ ) ، عن كُرَاع .<br>\
( و ) اسم ( صَنَم ، ويُخَفَّف ) ، وهو الأَشْهَر .<br>\
قال الأَزْهَرِيّ : وهو صَنَمٌ كانَت العَرَبُ تَنْصِبُه يَجَلون موضِعاً حوْلَه يَدُورون به ، واسمُ ذالِك الصَّنَمِ والموضعِ الدُّوَّارِ .<br>\
ومنه قولُ امرْىءِ القَيْسِ :<br>\
فَعَنَّ لنَا سِرْبٌ كأَنّ نِعَاجَهُ<br>\
عَذَارَى دُوَارٍ في مُلاَءٍ مُذَيَّلِ<br>\
أَراد بالسِّربِ البَقرَ ، ونِعَاجُه إِناثُه شَبَّهَها في مَشْيِها وطُولِ أَذنابها بجَوارٍ يَدُرْن حَولَ صَنَمٍ وعليهن المُلاَءُ المُذَيَّل ، أَي الطَّوِيل المُهَدَّب .<br>\
قال شيخُنَا : وقيل : إِنَّهُم كانُوا يَدُورُون حوَله أَسابِيعَ كما يُطَاف بالكَعْبَة .<br>\
ونقل الخَفَاجِيّ عن ابن الأَنباريّ : حِجَارَةٌ كانوا يَدُورُون حَوْلَهَا تَشْبِيهاً بالطائِفين بالكَعْبَة ، ولِذَا كَرِه الزَّمَخْشَرِيُّ وغيرُه أَن يُقَال دَارَ بالبَيْت ، بل يُقال : طافَ به .<br>\
( و ) الدَّوَّارَةُ ، ( كجَبَّانةٍ : الفِرْجَارُ ) ، وهو بالفارسيّة بركار ، وهي من أدواتِ النَّقَّاش والنَّجّار ، لها شُعْبَتَانِ يَنْضَمّان ويَنْفَرِجان لتَقْدِيرِ الدَّارَات .<br>\
( و ) الدُّوَّارُ ، ( بالضّمّ ، مُسْتَدَارُ رَمْل يَدُورُ حَوْلَهُ الوَحْشُ ) .<br>\

فمَا مُغْزِلٌ أَدْمَاءُ نَامَ غَزَالُهَا<br>\
بدُوَّارِ نِهْيٍ ذِي عَرَارٍ وحُلَّبِ<br>\
بأَحْسَنَ مِن لَيْلَة ولا أُمُّ شادِنٍ<br>\
غَضِيضَةُ طَرْفٍ رُعْتُهَا وَسْطَ رَبْرَبِ<br>\
( و ) عن ابن الأَعرابيّ : ( يقال لكلِّ ما لم يَتَحَرَّكْ ولم يَدُرْ : دَوَّارَةٌ وفوَّارَةٌ ) ، أَي ( بفَتْحِهما ، فإِذا تَحَرَّكَ أَو دَارَ ) ونَصُّ النّوادر : ودَار ( فهو دُوَّارَة وفُوَّارَة ) ، أَي ( بضمِّهما ) .<br>\
( والدّائِرَةُ : الحَلْقَةُ ) أَو شِبْهُهَا أَو الشَّيْءُ المُسْتَدِير .<br>\
( و ) الدَّائِرَةُ : ( الشَّعرُ المُسْتَدِيرُ على قَرْنِ الإِنْسَانِ ) .<br>\
ومن أَمثالهم ( ما اقشَعَرَّت له -دَائرتِي ) يُضرَب مَثَلاً لمن يَتَهَدَّدُكَ بالأَمر لا يَضُرُّك ، ( أَو ) الدّائرة : ( مَوْضعُ الذَّؤَابَةِ ) ، قاله ابنُ الأَعرابيّ .<br>\
( و ) الدَّائِرة : ( الهَزُيمَةُ ) والسُّوءُ .<br>\
يقال : { عَلَيْهِمْ دَائِرَةُ السَّوْء } وقوله تعالى : { نَخْشَى أَن تُصِيبَنَا دَآئِرَةٌ } قال أَبو عُبَيْدة أَي دَوْلة ، والدَّوائر تَدُور والدَّوائِلُ تَدُول .<br>\
( و ) الدَّائِرة ( الْتِي تَحْتَ الأَنْفِ ) يقال لها الدِّيرة ، والدَّائِرة ( كالدَّوّارَةِ ) ، بالتشديد .<br>\
( -والدَّارِيُّ : العَطَّارُ ) .<br>\
يقال : إِنه ( مَنْسُوبٌ أَلى دَارِينَ فُرْضَةٍ بالبَحْرَيْنه بها سُوقٌ ) كان ( يُحْمَلُ المِسْكُ من ) أَرض ( الهِنْدِ إِليها ) .<br>\
وقال الجَعْدِيّ :<br>\
أَلْقِيَ فِيها فِلْجَانِ من مِسْكِ دَا<br>\
رِينَ وفِلْجٌ من فُلْفُلٍ ضَرِمِ<br>\
وسأَلَ كِسْرَى عن دَارِينَ مَتَى كانَت ؟ فلم يَجِد أَحَداً يُخْبِره عنها إِلّا أَنَّهم اقلوا هي عَتِيقَةٌ بالفارسيّة فسُمِّيَت بها .<br>\
وفي الحديث : ( مَثَلُ الجَلِيس الصَّالِح مَثَلُ- الدّارِيّ إِن لم يُحْذِك من عِطْرِه عَلِقَك من رِيحِه ) .<br>\

( و ) الدَّارِيّ : ( رَبُّ النَّعَمِ ) ، سُمِّيَ بذالك لأَنه مُقِيم في دَارِه ، فنُسِب إِليها .<br>\
( و ) الدَّارِيُّ : ( المَلَّاحُ الذي يَلِي الشِّرَاعَ ) ، أَي القِلعَ .<br>\
( و ) الدّارِيُّ : ( اللازمُ لدِارِه ) لا يَبْرَح ولا يَطلُب مَعَاشاً ، ( كالدَّارِيَّةِ ) .<br>\
( و ) الدَّارِيُّ ( مِنَ الإِبِل : المُتَخلِّفُ في مَبْرَكِه ) لا يَخْرُج إِلى المَرْعَى ، وكذالك شاةٌ دَارِيَّةٌ .<br>\
( والمُدَاوَرَةُ كالمُعَالَجَة ) في الأُمور ، وهو طَلَبُ وُجُوهِ مَأْتَاهَا ، وهو<br>\
